204  template <typename T>
205  string Serialization::vectorTag(const string& tagName, const vector<T>& values,
206  const NameMap& idMap, const string& elementTagName,
207  T defaultValue,int numTab)
208  {
209  ostringstream ss;
210  ss << getTabs(numTab);
211 
212  // The header of the vector
213  ss << "<" << tagName << " size=\"" << values.size() << "\">\n";
214 
215  // save the default value
216  ss << Serialization::standardTag("default", defaultValue, numTab+1) << endl;
217 
218  // The vector values
219  for (int i = 0; i < (int)values.size(); ++i)
220  {
221  if (values[i] != defaultValue) {
222  ss << getTabs(numTab+1);
223  ss << "<" << elementTagName << " id=\"" << idMap.getNameFromIdx(i) << "\">"
224  << values[i] << "</" << elementTagName << ">\n";
225  }
226  }
227 
228  // closing tag
229  ss << getTabs(numTab);
230  ss << "</" << tagName << ">"; // close tag
231 
232  return ss.str();
233  }

444  template <typename T>
445  T UnSerialization::seekAndParseEnclosedValue(nor_utils::StreamTokenizer& st, const string& tag)
446  {
447  do {
448  // keeps getting the next token until it finds the tag
449  if ( nor_utils::cmp_nocase( st.next_token(), tag ) )
450  {
451  T val;
452  istringstream ss(st.next_token());
453  ss >> val;
454  return val;
455  }
456  } while( st.has_token() );
457 
458  return T();
459  }

463  template <typename T>
464  void UnSerialization::seekAndParseVectorTag(nor_utils::StreamTokenizer& st,
465  const string& arrayName,
466  const NameMap& idMap,
467  const string& elementTagName,
468  vector<T>& vecToFill)
469  {
470 
471  string tagParam, paramVal;
472 
473  // move until the tag arrayName, then get the value of the param "size"
474  seekAndParseParamTag(st, arrayName, tagParam, paramVal);
475 
476  // get the default value (sparse represenation)
477  T defaultValue = seekAndParseEnclosedValue<T>(st, "default");;
478 
479  // resize the vector to vector size
480  // (FIXME: not very safe!)
481  vecToFill.resize( atoi(paramVal.c_str() ) );
482 
483  // fill the vector with the default value
484  fill(vecToFill.begin(), vecToFill.end(), defaultValue);
485 
486  T tmpVal;
487  string className;
488  string tag, param;
489 
490  // parses the xml vector definition
491  const string closeTag = "/" + arrayName;
492  while ( !nor_utils::cmp_nocase(tag, closeTag) && st.has_token() )
493  {
494  // seek the parameter "class"
495  parseParamTag(st.next_token(), tag, param, paramVal );
496  if ( !nor_utils::cmp_nocase(tag, elementTagName) )
497  continue;
498 
499  // convert the string into a value
500  istringstream ss( st.next_token() );
501  ss >> tmpVal;
502 
503  // stores the value, using the mapping from the class name
504  // to the class index.
505  vecToFill[ idMap.getIdxFromName(paramVal) ] = tmpVal;
506  }
507 
508  }
