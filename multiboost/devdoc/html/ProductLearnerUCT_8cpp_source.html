71  void ProductLearnerUCT::initLearningOptions(const nor_utils::Args& args)
72  {
73  BaseLearner::initLearningOptions(args);
74 
75  string baseLearnerName;
76  args.getValue("baselearnertype", 0, baseLearnerName);
77  args.getValue("baselearnertype", 1, _numBaseLearners);
78 
79  // get the registered weak learner (type from name)
80  BaseLearner* pWeakHypothesisSource =
81  BaseLearner::RegisteredLearners().getLearner(baseLearnerName);
82 
83  for( int ib = 0; ib < _numBaseLearners; ++ib ) {
84  _baseLearners.push_back(pWeakHypothesisSource->create());
85  _baseLearners[ib]->initLearningOptions(args);
86  }
87 
88  string updateRule = "";
89  if ( args.hasArgument( "updaterule" ) )
90  args.getValue("updaterule", 0, updateRule );
91 
92  if ( updateRule.compare( "edge" ) == 0 )
93  _updateRule = EDGE_SQUARE;
94  else if ( updateRule.compare( "alphas" ) == 0 )
95  _updateRule = ALPHAS;
96  else if ( updateRule.compare( "edgesquare" ) == 0 )
97  _updateRule = ESQUARE;
98  else {
99  cerr << "Unknown update rule in ProductLearnerUCT (set to default [edge]" << endl;
100  _updateRule = EDGE_SQUARE;
101  }
102  }

106  AlphaReal ProductLearnerUCT::classify(InputData* pData, int idx, int classIdx)

116  AlphaReal ProductLearnerUCT::run()
117  {
118  if ( _numOfCalling == 0 ) {
119  if (_verbose > 0) {
120  cout << "Initializing tree..." << endl;
121  }
122  InnerNodeUCTSparse::setDepth( _numBaseLearners );
123  InnerNodeUCTSparse::setBranchOrder( _pTrainingData->getNumAttributes() );
124  _root.setChildrenNum();
125  //createUCTTree();
126  }
127  _numOfCalling++;
128 
129  const int numClasses = _pTrainingData->getNumClasses();
130  const int numExamples = _pTrainingData->getNumExamples();
131 
132  // Backup original labels
133  for (int i = 0; i < numExamples; ++i) {
134  const vector<Label>& labels = _pTrainingData->getLabels(i);
135  vector<char> exampleLabels;
136  for (int l = 0; l < numClasses; ++l)
137  exampleLabels.push_back(labels[l].y);
138  _savedLabels.push_back(exampleLabels);
139  }
140 
141  for(int ib = 0; ib < _numBaseLearners; ++ib)
142  _baseLearners[ib]->setTrainingData(_pTrainingData);
143 
144  AlphaReal energy = numeric_limits<float>::max();
145  AlphaReal previousEnergy, hx, previousAlpha;
146  BaseLearner* pPreviousBaseLearner = 0;
147 
148  vector< int > trajectory(0);
149  _root.getBestTrajectory( trajectory );
150 
151  if (_verbose > 2) {
152  cout << "Trajectory: ";
153  for( int i = 0; i < trajectory.size(); i++ ) {
154  cout << trajectory[i] << " ";
155  }
156 
157  cout << endl;
158  }
159 
160 
161  bool firstLoop = true;
162  int ib = -1;
163  while (1) {
164  ib += 1;
165  if (ib >= _numBaseLearners) {
166  ib = 0;
167  firstLoop = false;
168  }
169  previousEnergy = energy;
170  previousAlpha = _alpha;
171  if (pPreviousBaseLearner)
172  delete pPreviousBaseLearner;
173  if ( !firstLoop ) {
174  // take the old learner off the labels
175  for (int i = 0; i < numExamples; ++i) {
176  vector<Label>& labels = _pTrainingData->getLabels(i);
177  for (int l = 0; l < numClasses; ++l) {
178  // Here we could have the option of using confidence rated setting so the
179  // real valued output of classify instead of its sign
180  hx = _baseLearners[ib]->classify(_pTrainingData,i,l);
181  if ( hx < 0 )
182  labels[l].y *= -1;
183  else if ( hx == 0 ) { // have to redo the multiplications, haven't been tested
184  for(int ib1 = 0; ib1 < _numBaseLearners && labels[l].y != 0; ++ib1) {
185  if (ib != ib1) {
186  hx = _baseLearners[ib1]->classify(_pTrainingData,i,l);
187  if (hx < 0)
188  labels[l].y *= -1;
189  else if (hx == 0)
190  labels[l].y = 0;
191  }
192  }
193  }
194  }
195  }
196  }
197  pPreviousBaseLearner = _baseLearners[ib]->copyState();
198  energy = dynamic_cast<FeaturewiseLearner*>(_baseLearners[ib])->run( trajectory[ib] );
199  _alpha = _baseLearners[ib]->getAlpha();
200  if (_verbose > 2) {
201  cout << "E[" << (ib+1) << "] = " << energy << endl << flush;
202  cout << "alpha[" << (ib+1) << "] = " << _alpha << endl << flush;
203  }
204  for (int i = 0; i < numExamples; ++i) {
205  vector<Label>& labels = _pTrainingData->getLabels(i);
206  for (int l = 0; l < numClasses; ++l) {
207  // Here we could have the option of using confidence rated setting so the
208  // real valued output of classify instead of its sign
209  if (labels[l].y != 0) { // perhaps replace it by nor_utils::is_zero(labels[l].y)
210  hx = _baseLearners[ib]->classify(_pTrainingData,i,l);
211  if ( hx < 0 )
212  labels[l].y *= -1;
213  else if ( hx == 0 )
214  labels[l].y = 0;
215  }
216  }
217  }
218 
219  // We have to do at least one full iteration. For real it's not guaranteed
220  // Alternatively we could initialize all of them to constant
221  // if ( !firstLoop && energy >= previousEnergy ) {
222  // if (energy > previousEnergy) {
223  // _baseLearners[ib] = pPreviousBaseLearner->copyState();
224  // delete pPreviousBaseLearner;
225  // energy = previousEnergy;
226  // _alpha = _baseLearners[ib]->getAlpha();
227  // }
228  // break;
229  // }
230  if ( energy >= previousEnergy ) {
231  _alpha = previousAlpha;
232  energy = previousEnergy;
233  if (firstLoop) {
234  for(int ib2 = ib; ib2 < _numBaseLearners; ++ib2)
235  delete _baseLearners[ib2];
236  _numBaseLearners = ib;
237  }
238  else {
239  _baseLearners[ib] = pPreviousBaseLearner->copyState();
240  }
241  delete pPreviousBaseLearner;
242  break;
243  }
244  }
245 
246  // Restore original labels
247  for (int i = 0; i < numExamples; ++i) {
248  vector<Label>& labels = _pTrainingData->getLabels(i);
249  for (int l = 0; l < numClasses; ++l)
250  labels[l].y = _savedLabels[i][l];
251  }
252 
253 
254  //update the weights in the UCT tree
255  double updateWeight = 0.0;
256  if ( _updateRule == EDGE_SQUARE ) {
257  double edge = getEdge();
258  updateWeight = 1 - sqrt( 1 - ( edge * edge ) );
259  } else if ( _updateRule == ALPHAS ) {
260  double alpha = this->getAlpha();
261  updateWeight = alpha;
262  } else if ( _updateRule == ESQUARE ) {
263  double edge = getEdge();
264  updateWeight = edge * edge;
265  }
266 
267 
268  _root.updateInnerNodes( updateWeight, trajectory );
269 
270  if (_verbose > 2) {
271  cout << "Update weight (" << updateWeight << ")" << "\tUpdate rule index: "<< _updateRule << endl << flush;
272  }
273 
274  _id = _baseLearners[0]->getId();
275  for(int ib = 1; ib < _numBaseLearners; ++ib)
276  _id += "_x_" + _baseLearners[ib]->getId();
277  return energy;
278  }
