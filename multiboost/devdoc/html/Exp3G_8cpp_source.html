68  int Exp3G::getNextAction()
69  {
70  vector< AlphaReal > cumsum( getArmNumber()+1 );
71  int i;
72 
73  cumsum[0] = 0.0;
74  for( int i=0; i < getArmNumber(); i++ )
75  {
76  cumsum[i+1] = cumsum[i] + _p[i];
77  }
78 
79  for( i=0; i < getArmNumber(); i++ )
80  {
81  cumsum[i+1] /= cumsum[ getArmNumber() ];
82  }
83 
84  AlphaReal r = rand() / (AlphaReal) RAND_MAX;
85 
86  for( i=0; i < getArmNumber(); i++ )
87  {
88  if ( (cumsum[i] <= r) && ( r<=cumsum[i+1] ) ) break;
89  }
90 
91  return i;
92  }

154  for( int i=0; i<_numOfArms; i++ )

179  void Exp3G::receiveReward( int armNum, AlphaReal reward )
180  {
181  _T[ armNum ]++;
182  // calculate the feedback value
183 
184  int prevArm;
185  if ( _actions.size() >= 1 )
186  {
187  prevArm = _actions.back();
188  _actions.push_back( armNum );
189  _sideInformation[armNum][prevArm]++;
190  } else {
191  _sideInformation[armNum][armNum]++;
192  _actions.push_back( armNum );
193  prevArm = armNum;
194  }
195  incIter();
196 
197  //update
198  for( int i=0; i < _numOfArms; i++ )
199  {
200  _w[i] += ( ( ( (AlphaReal)_sideInformation[i][prevArm] ) / ( (AlphaReal)_sideInformation[armNum][prevArm] ) ) * ( _eta * reward ) );
201  }
202 
203  //_w[armNum] += ( _eta * reward );
204 
205  updateithValue( armNum );
206  }
