77  void StochasticLearner::initLearningOptions(const nor_utils::Args& args)
78  {
79  BaseLearner::initLearningOptions(args);
80 
81  if (args.hasArgument("initgamma"))
82  args.getValue("initgamma", 0, _initialGammat);
83 
84  if (args.hasArgument("gammdivperiod"))
85  args.getValue("gammdivperiod", 0, _gammdivperiod);
86 
87 
88  if (args.hasArgument("graditer"))
89  args.getValue("graditer", 0, _maxIter);
90 
91  if (args.hasArgument("gradmethod"))
92  {
93  string gradMethod;
94  args.getValue("gradmethod", 0, gradMethod);
95 
96  if ( gradMethod.compare( "sgd" ) == 0 )
97  _gMethod = OPT_SGD;
98  else if ( gradMethod.compare( "bgd" ) == 0 )
99  _gMethod = OPT_BGD;
100  else {
101  cerr << "SigmoidSingleStumpLearner::Unknown update gradient method" << endl;
102  exit( -1 );
103  }
104  }
105 
106  if (args.hasArgument("tfunc"))
107  {
108  string targetFunction;
109  args.getValue("tfunc", 0, targetFunction);
110 
111  if ( targetFunction.compare( "exploss" ) == 0 )
112  _tFunction = TF_EXPLOSS;
113  else if ( targetFunction.compare( "edge" ) == 0 )
114  _tFunction = TF_EDGE;
115  else {
116  cerr << "SigmoidSingleStumpLearner::Unknown target function" << endl;
117  exit( -1 );
118  }
119 
120  }
121 
122  }
