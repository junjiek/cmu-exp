47  AlphaReal AdaLineLearner::run()
48  {
49  const int numClasses = _pTrainingData->getNumClasses();
50  const int numColumns = _pTrainingData->getNumAttributes();
51  const int numExamples = _pTrainingData->getNumExamples();
52 
53  vector<AlphaReal> bestv(numClasses);
54 
55  _v.resize(numClasses);
56 
57  AlphaReal gammat = _initialGammat;
58 
59  if (_verbose>4)
60  cout << "-->Init gamma: " << gammat << endl;
61 
62  // set the smoothing value to avoid numerical problem
63  // when theta=0.
64  setSmoothingVal( 1.0 / (AlphaReal)_pTrainingData->getNumExamples() * 0.01 );
65 
66  _featuresWeight.resize(numColumns);
67  fill(_featuresWeight.begin(),_featuresWeight.end(), 1.0 );
68  nor_utils::normalizeLengthOfVector(_featuresWeight);
69 
70  _v.resize( numClasses );
71  for(int i=0; i<numClasses; ++i )
72  {
73  _v[i] = (rand() % 2) * 2 - 1;
74  //fill(vsArray[i].begin(),vsArray[i].end(),0.0);
75  }
76  nor_utils::normalizeLengthOfVector( _v );
77 
78 
79  if ( _gMethod == OPT_SGD )
80  {
81  vector<int> randomPermutation(numExamples);
82  for (int i = 0; i < numExamples; ++i ) randomPermutation[i]=i;
83  random_shuffle( randomPermutation.begin(), randomPermutation.end() );
84 
85  AlphaReal gammaDivider = 1.0;
86  for (int i = 0; i < numExamples; ++i )
87  {
88  if ((i>0)&&((i%_gammdivperiod)==0)) gammaDivider += 1.0;
89 
90  int randomTrainingInstanceIdx = randomPermutation[i];
91  vector<Label> labels = _pTrainingData->getLabels(randomTrainingInstanceIdx);
92 
93  FeatureReal innerProduct = 0.0;
94  for (int j = 0; j < numColumns; ++j)
95  {
96  innerProduct += _featuresWeight[j] * _pTrainingData->getValue(randomTrainingInstanceIdx, j);
97  }
98 
99  vector<AlphaReal> deltaV(numClasses, 0.0);
100  switch (_tFunction) {
101  case TF_EXPLOSS:
102  for( vector< Label >::iterator it = labels.begin(); it != labels.end(); it++ )
103  {
104  deltaV[it->idx] = it->weight * exp( - _v[it->idx] * it->y * innerProduct ) *
105  it->y * innerProduct;
106  }
107  break;
108  case TF_EDGE:
109  // has to be implemented
110  break;
111  default:
112  break;
113  }
114 
115 
116 
117  vector<AlphaReal> deltaW(numColumns, 0.0);
118  for (int j = 0; j < numColumns; ++j)
119  {
120  FeatureReal val = _pTrainingData->getValue(randomTrainingInstanceIdx, j);
121 
122  switch (_tFunction) {
123  case TF_EXPLOSS:
124  for( vector< Label >::iterator it = labels.begin(); it != labels.end(); it++ )
125  {
126  deltaW[j] += (it->weight * exp( - _v[it->idx] * it->y * innerProduct ) *
127  val * _v[it->idx] * it->y );
128  }
129  break;
130  case TF_EDGE:
131  // has to be implemented
132  break;
133  default:
134  break;
135  }
136  }
137 
138  // gradient step
139  for (int j = 0; j < numColumns; ++j)
140  {
141  _featuresWeight[j] -= numExamples * static_cast<FeatureReal>(gammat * deltaW[j]);
142  }
143  nor_utils::normalizeLengthOfVector( _featuresWeight );
144 
145  for (int j = 0; j < numClasses; ++j)
146  {
147  _v[j] -= numExamples * static_cast<FeatureReal>(gammat * deltaV[j]);
148  }
149  nor_utils::normalizeLengthOfVector( _v );
150 
151  }
152 
153  } else if (_gMethod == OPT_BGD )
154  {
155  } else {
156  cout << "Unknown optimization method!" << endl;
157  exit(-1);
158  }
159 
160  for(int k=0; k<numClasses; ++k ) _v[k]= _v[k] < 0 ? -1.0 : 1.0;
161 
162  //calculate alpha
163  this->_alpha = 0.0;
164  AlphaReal eps_min = 0.0, eps_pls = 0.0;
165 
166  //_pTrainingData->clearIndexSet();
167  for( int i = 0; i < _pTrainingData->getNumExamples(); i++ )
168  {
169  vector< Label> l = _pTrainingData->getLabels( i );
170  for( vector< Label >::iterator it = l.begin(); it != l.end(); it++ ) {
171  AlphaReal result = this->classify( _pTrainingData, i, it->idx );
172  result *= (it->y * it->weight);
173  if ( result < 0 ) eps_min -= result;
174  if ( result > 0 ) eps_pls += result;
175  }
176 
177  }
178 
179  this->_alpha = getAlpha( eps_min, eps_pls );
180 
181  if (_verbose>2)
182  {
183  cout << "---> Alpha: " << this->_alpha << endl;
184  }
185 
186  AlphaReal edge = this->getEdge(true);
187 
188  return edge;
189  }
