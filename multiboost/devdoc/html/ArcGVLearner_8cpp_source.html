66  void ArcGVLearner::run(const nor_utils::Args& args)
67  {
68  // load the arguments
69  this->getArgs(args);
70 
71  // get the registered weak learner (type from name)
72  BaseLearner* pWeakHypothesisSource =
73  BaseLearner::RegisteredLearners().getLearner(_baseLearnerName);
74  // initialize learning options; normally it's done in the strong loop
75  // also, here we do it for Product learners, so input data can be created
76  pWeakHypothesisSource->initLearningOptions(args);
77 
78  BaseLearner* pConstantWeakHypothesisSource =
79  BaseLearner::RegisteredLearners().getLearner("ConstantLearner");
80 
81  // get the training input data, and load it
82 
83  InputData* pTrainingData = pWeakHypothesisSource->createInputData();
84  pTrainingData->initOptions(args);
85  pTrainingData->load(_trainFileName, IT_TRAIN, _verbose);
86 
87  // get the testing input data, and load it
88  InputData* pTestData = NULL;
89  if ( !_testFileName.empty() )
90  {
91  pTestData = pWeakHypothesisSource->createInputData();
92  pTestData->initOptions(args);
93  pTestData->load(_testFileName, IT_TEST, _verbose);
94  }
95 
96  // The output information object
97  OutputInfo* pOutInfo = NULL;
98 
99  if ( !_outputInfoFile.empty() )
100  {
101  pOutInfo = new OutputInfo(args);
102  pOutInfo->initialize(pTrainingData);
103 
104  if (pTestData)
105  pOutInfo->initialize(pTestData);
106  pOutInfo->outputHeader(pTrainingData->getClassMap());
107 
108  if ( ! args.hasArgument("resume") )
109  {
110  // Baseline: constant classifier - goes into 0th iteration
111 
112  BaseLearner* pConstantWeakHypothesis = pConstantWeakHypothesisSource->create() ;
113  pConstantWeakHypothesis->initLearningOptions(args);
114  pConstantWeakHypothesis->setTrainingData(pTrainingData);
115  pConstantWeakHypothesis->run();
116 
117  pOutInfo->outputIteration(-1);
118  pOutInfo->outputCustom(pTrainingData, pConstantWeakHypothesis);
119  if (pTestData != NULL)
120  {
121  pOutInfo->separator();
122  pOutInfo->outputCustom(pTestData, pConstantWeakHypothesis);
123  }
124  pOutInfo->outputCurrentTime();
125  pOutInfo->endLine();
126 
127  pOutInfo->initialize(pTrainingData);
128  if (pTestData)
129  pOutInfo->initialize(pTestData);
130  }
131  } else {
132  cout << "Output file is empty!!!!" << endl;
133  exit(-1);
134  }
135 
136  //cout << "Before serialization" << endl;
137  // reload the previously found weak learners if -resume is set.
138  // otherwise just return 0
139  int startingIteration = resumeWeakLearners(pTrainingData);
140 
141  Serialization ss(_shypFileName, _isShypCompressed );
142  ss.writeHeader(_baseLearnerName); // this must go after resumeProcess has been called
143 
144  // perform the resuming if necessary. If not it will just return
145  resumeProcess(ss, pTrainingData, pTestData, pOutInfo);
146 
147 
148  if (_verbose == 1)
149  cout << "Learning in progress..." << endl;
150 
151  //I put here the starting time, but it may take very long time to load the saved model
152  time_t startTime, currentTime;
153  time(&startTime);
154 
156  // Starting the AdaBoost main loop
158  for (int t = startingIteration; t < _numIterations; ++t)
159  {
160  if (_verbose > 1)
161  cout << "------- WORKING ON ITERATION " << (t+1) << " -------" << endl;
162 
163  BaseLearner* pWeakHypothesis = pWeakHypothesisSource->create();
164  pWeakHypothesis->initLearningOptions(args);
165  //pTrainingData->clearIndexSet();
166 
167  pWeakHypothesis->setTrainingData(pTrainingData);
168 
169  AlphaReal energy = pWeakHypothesis->run();
170 
171  //float gamma = pWeakHypothesis->getEdge();
172  //cout << gamma << endl;
173 
174  if ( (_withConstantLearner) || ( energy != energy ) ) // check constant learner if user wants it (if energi is nan, then we chose constant learner
175  {
176  BaseLearner* pConstantWeakHypothesis = pConstantWeakHypothesisSource->create() ;
177  pConstantWeakHypothesis->initLearningOptions(args);
178  pConstantWeakHypothesis->setTrainingData(pTrainingData);
179  AlphaReal constantEnergy = pConstantWeakHypothesis->run();
180 
181  if ( (constantEnergy <= energy) || ( energy != energy ) ) {
182  delete pWeakHypothesis;
183  pWeakHypothesis = pConstantWeakHypothesis;
184  }
185  }
186 
187  // find the min margin on training data
188  table& g = pOutInfo->getMargins( pTrainingData );
189  AlphaReal alphaSum = pOutInfo->getSumOfAlphas(pTrainingData);
190  if (nor_utils::is_zero(alphaSum)) alphaSum = 1.0;
191 
192  AlphaReal minMargin = numeric_limits<AlphaReal>::max();
193  for (int i=0; i<pTrainingData->getNumExamples(); ++i )
194  {
195  const vector<Label>& labels = pTrainingData->getLabels(i);
196  vector<Label>::const_iterator lIt;
197 
198  for (lIt = labels.begin(); lIt != labels.end(); ++lIt )
199  {
200  AlphaReal tmpMargin = g[i][lIt->idx] / alphaSum;
201  if ( minMargin > tmpMargin) minMargin = tmpMargin;
202  }
203  }
204 
205  if (_verbose>2)
206  {
207  cout << "---> Min margin: " << minMargin << endl;
208  }
209 
210  // add the arc-gv term to the coefficient of base classifier
211  // if the minMargin is smaller than a threshold, it is set to a user-defined value
212  if (minMargin< _minMarginThreshold) minMargin = _minMarginThreshold;
213 
214  // update alpha
215  AlphaReal alpha = pWeakHypothesis->getAlpha();
216  AlphaReal newAlpha = alpha - 0.5 * log ( ( 1.0 + minMargin ) / ( 1.0 - minMargin ) );
217  pWeakHypothesis->setAlpha(newAlpha);
218 
219  // update the sum of coefficients
220  _alphaSum+=newAlpha;
221 
222  if (_verbose>2)
223  {
224  cout << "---> Alpha (based on AdaBoost.MH): " << alpha << endl;
225  cout << "---> Alpha (based on ARC-GV): " << newAlpha << endl << flush;
226  }
227 
228 
229  if (_verbose > 1)
230  cout << "Weak learner: " << pWeakHypothesis->getName()<< endl;
231 
232 
233  // Output the step-by-step information
234  printOutputInfo(pOutInfo, t, pTrainingData, pTestData, pWeakHypothesis);
235 
236  // Updates the weights and returns the edge, and update the alpha
237  AlphaReal gamma = updateWeights(pTrainingData, pWeakHypothesis);
238 
239 
240  if (_verbose > 1)
241  {
242  cout << setprecision(5)
243  << "--> Alpha = " << pWeakHypothesis->getAlpha() << endl
244  << "--> Edge = " << gamma << endl
245  << "--> Energy = " << energy << endl
246  // << "--> ConstantEnergy = " << constantEnergy << endl
247  // << "--> difference = " << (energy - constantEnergy) << endl
248  ;
249  }
250 
251  // If gamma <= theta the algorithm must stop.
252  // If theta == 0 and gamma is 0, it means that the weak learner is no better than chance
253  // and no further training is possible.
254  if (gamma <= _theta)
255  {
256  if (_verbose > 0)
257  {
258  cout << "Can't train any further: edge = " << gamma
259  << " (with and edge offset (theta)=" << _theta << ")" << endl;
260  }
261 
262  // delete pWeakHypothesis;
263  // break;
264  }
265 
266  // append the current weak learner to strong hypothesis file,
267  // that is, serialize it.
268  ss.appendHypothesis(t, pWeakHypothesis);
269 
270  // Add it to the internal list of weak hypotheses
271  _foundHypotheses.push_back(pWeakHypothesis);
272 
273  // check if the time limit has been reached
274  if (_maxTime > 0)
275  {
276  time( &currentTime );
277  float diff = difftime(currentTime, startTime); // difftime is in seconds
278  diff /= 60; // = minutes
279 
280  if (diff > _maxTime)
281  {
282  if (_verbose > 0)
283  cout << "Time limit of " << _maxTime
284  << " minutes has been reached!" << endl;
285  break;
286  }
287  } // check for maxtime
288  delete pWeakHypothesis;
289  } // loop on iterations
291 
292  // write the footer of the strong hypothesis file
293  ss.writeFooter();
294 
295  // write the weights of the instances if the name of weights file isn't empty
296  printOutWeights( pTrainingData );
297 
298 
299  // Free the two input data objects
300  if (pTrainingData)
301  delete pTrainingData;
302  if (pTestData)
303  delete pTestData;
304 
305  if (pOutInfo)
306  delete pOutInfo;
307 
308  if (_verbose > 0)
309  cout << "Learning completed." << endl;
310  }
