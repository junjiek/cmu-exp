162  inline FeatureReal getValue(int idx, int columnIdx) const {
163  if ( _dataRep == DR_DENSE ) return _data[idx].getValues()[columnIdx];
164  else {
165  map<int,int>::iterator it = ((Example &)_data[idx]).getValuesIndexesMap().find( columnIdx );
166  if ( it == ((Example &)_data[idx]).getValuesIndexesMap().end() ) return 0;
167  else return _data[idx].getValues()[it->second];
168  }
169  }

178  void addExample( Example example ) {
179  _data.push_back( example );
180  _numExamples++;
181 
182  const vector<Label> & labels = example.getLabels();
183  vector<Label>::const_iterator lIt;
184 
185  for (lIt = labels.begin(); lIt != labels.end(); ++lIt) {
186  if ( lIt->y > 0) {
187  _nExamplesPerClass[lIt->idx]++;
188  }
189  }
