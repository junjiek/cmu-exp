49  void HaarLearner::declareArguments(nor_utils::Args& args)
50  {
51  args.declareArgument("ftypes",
52  "Specify the list with the type of Haar-like features. This list can contain "
53  "all the types requested at once. The types allowed are:\n"
54  " 2v: 2 blocs vertical feature\n"
55  " 2h: 2 blocs horizontal feature\n"
56  " 3v: 3 blocs vertical feature\n"
57  " 3h: 3 blocs horizontal feature\n"
58  " 4q: 4 blocs squared feature\n"
59  "For instance if you need all the 2 and 3 blocs type, just write:\n"
60  "-ftypes 2v2h3v3h\n"
61  "(DEFAULT: ALL of them will be selected)"
62  , 1, "<list>");
63 
64  args.declareArgument("csample",
65  "Instead of computing all the possible combinations, select a set of "
66  "random configurations. It is possible to specify the number of combinations "
67  "or the amount of time per feature type. The options are:\n"
68  " num: The number of combination per type per iteration.\n"
69  " time: The amount of time (in seconds) per type per iteration.\n"
70  "Example: -csample num 4 -> Set the number of random configurations to 4.\n"
71  "Example: -csample time 120 -> Set the time to 2 minutes per type per iteration.",
72  2, "<opt> <#|sec>");
73 
74  args.declareArgument("iisize",
75  "Specify the size of the integral image representation for each example,\n"
76  "Example: -iisize 128x64",
77  1, "<width>x<height>");
78 
79  }

83  void HaarLearner::initOptions(const nor_utils::Args& args)
84  {
85  // If the sampling is required
86  if (args.hasArgument("csample"))
87  {
88  string randSearch = args.getValue<string>("csample", 0);
89  // if the type is ST_NUM this is the number of samplings per feature per iteration
90  // if the type is ST_TIME this is the amount of time per feature per iteration
91  _samplingVal = args.getValue<int>("csample", 1);
92 
93  if (randSearch == "num")
94  _samplingType = ST_NUM;
95  else if (randSearch == "time")
96  _samplingType = ST_TIME;
97  }
98 
99  }

120  void HaarLearner::load(nor_utils::StreamTokenizer& st)
121  {
122  // get the type of feature
123  string shortName = UnSerialization::seekAndParseEnclosedValue<string>(st, "type");
124  // get it from the registered features
125  _pSelectedFeature = HaarFeature::RegisteredFeatures().getFeature(shortName);
126 
127  // Load the coordinates and size of the feature
128  _selectedConfig.x = UnSerialization::seekAndParseEnclosedValue<short>(st, "x");
129  _selectedConfig.y = UnSerialization::seekAndParseEnclosedValue<short>(st, "y");
130  _selectedConfig.width = UnSerialization::seekAndParseEnclosedValue<short>(st, "width");
131  _selectedConfig.height = UnSerialization::seekAndParseEnclosedValue<short>(st, "height");
132 
133  }
