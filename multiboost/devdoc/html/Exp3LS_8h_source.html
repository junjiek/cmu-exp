145  template< typename BaseType,typename KeyType>
146  KeyType Exp3LS<BaseType,KeyType>::getNextAction( KeyType defaultValue )
147  {
148  KeyType retVal=defaultValue;
149  double r = rand() / (double) RAND_MAX;
150  if ( (! _cumSum.empty() ) && ( r < _cumSum.back().first ) )
151  {
152  for( itVectorPairBaseKey it = _cumSum.begin(); it != _cumSum.end(); it++ )
153  {
154  if ( r < it->first ) {
155  retVal = it->second;
156  break;
157  }
158  }
159  }
160 
161  return retVal;
162  }

166  template< typename BaseType,typename KeyType>
167  void Exp3LS<BaseType,KeyType>::initialize( map<KeyType,BaseType>& vals )
168  {
169  _p.clear();
170  //according to the paper cited in the header, we set this parameter sqrt( (KlnK)/(e-1)g ) based on the Corollary 3.2
171  //_alpha = pow( 4*_numOfArms*log((double)_numOfArms) * ( 1.0 / 100000 ), 1.0/3.0);
172  _gamma = pow( (this->getArmNumber()*log((double)this->getArmNumber()))/( 2.0 * 100000 ) , 1.0/3.0);
173  //if ( _gamma > 1.0 ) _gamma = 1.0;
174  _Xsum = static_cast<double>( 0.0 );
175 
176  _expsum = this->getArmNumber() * exp( 1.0 / this->getArmNumber() );
177 
178  for( itMapKeyBase it = vals.begin(); it != vals.end(); it++ )
179  {
180  receiveReward( it->first, it->second );
181  }
182 
183  this->setInitializedFlagToTrue();
184  }

188  template< typename BaseType,typename KeyType>
189  void Exp3LS<BaseType,KeyType>::updateithValue( KeyType key )
190  {
191  itMapKeyBase it;
192  _cumSum.resize( this->_X.size()+1 );
193  //_cumSumKeys.resize( _p.size()+1 );
194 
195  int i=0;
196  _cumSum[i].first = static_cast<BaseType>(0.0);
197 
198  BaseType pTemp;
199 
200  for( it=this->_X.begin(), i=0; it != this->_X.end(); it++, i++ )
201  {
202  pTemp = getPValue( key );
203  _cumSum[i+1].first = _cumSum[i].first + pTemp;
204  _cumSum[i+1].second = it->first;
205  }
206  //renormalize the probabilities
207 
208 
209  //because of the numerical inaccuracy we set the last element of the cummulative sum to one if all arms have already pulled at least once
210  if ( (_cumSum.size() - 1) == this->getArmNumber()) _cumSum[ _cumSum.size()- 1].first = static_cast<BaseType>(1.0);
211  }

215  template< typename BaseType,typename KeyType>
216  void Exp3LS<BaseType,KeyType>::receiveReward( KeyType key, BaseType reward )
217  {
218  BaseType xhat;
219  BaseType prevValue;
220 
221  // calculate the p value
222  xhat = getPValue( key );
223 
224  //_X[ key ] += static_cast<BaseType>( ( _gamma * (xhat / _numOfArms) ) );
225  if ( this->_X.find( key ) == this->_X.end() )
226  {
227  prevValue = 0.0;
228  this->_T[ key ]=1;
229  this->_X[ key ] = static_cast<BaseType>( ( _gamma / this->getArmNumber() ) * ( reward / xhat ) );
230  } else {
231  prevValue = this->_X[key];
232  this->_T[ key ]++;
233  this->_X[ key ] += static_cast<BaseType>( ( _gamma / this->getArmNumber() ) * ( reward / xhat ) );
234  }
235  //update the sum of the w values
236  BaseType deltaw = this->_X[ key ] - prevValue;
237  _Xsum += deltaw;
238 
239  this->incIter();
240  //update the probabilities of the arms
241  updateithValue( key );
242  }

244 //----------------------------------------------------------------
245 //----------------------------------------------------------------
246  template< typename BaseType,typename KeyType>
247  BaseType Exp3LS<BaseType,KeyType>::getPValue( KeyType key )
248  {
249  if ( this->_X.find( key ) == this->_X.end() ) return static_cast<BaseType>( 1 / (double)this->getArmNumber());
250  BaseType pValue=0.0;
251  if ( _Xsum > 0.0 )
252  {
253 
254  double tmpSum = ( this->getArmNumber() - this->_X.size() ) * (1.0 / this->getArmNumber());
255  tmpSum += _Xsum;
256  //double expSum = ( _numOfArms - _X.size() ) * exp(1.0 / _numOfArms);
257  //for( map<KeyType,BaseType>::iterator it = _X.begin(); it != _X.end(); it++ ) expSum += exp( it->second / tmpSum );
258 
259 
260  //pValue = static_cast<float>( ( 1 - _gamma ) * ( ( exp( _X[key] / tmpSum ) )/ _expSum ) + ( _gamma / _numOfArms ) );
261  pValue = static_cast<BaseType>( ( 1 - _gamma ) * ( ( ( this->_X[key] / tmpSum ) ) ) + ( _gamma / this->getArmNumber() ) );
262  } else {
263  pValue = static_cast<BaseType>( 1 / (double)this->getArmNumber());
264  }
265  return pValue;
266  }
