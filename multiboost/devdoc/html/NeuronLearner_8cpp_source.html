56  void NeuronLearner::save(ofstream& outputStream, int numTabs)
57  {
58  // Calling the super-class method
59  AbstainableLearner::save(outputStream, numTabs);
60 
61  // save feature weights
62  stringstream featureWeightsSS("");
63  vector<FeatureReal>::iterator it = _featuresWeight.begin();
64  featureWeightsSS << *it;
65  ++it;
66  for (;it != _featuresWeight.end(); ++it )
67  {
68  featureWeightsSS << " " << *it;
69  }
70 
71  outputStream << Serialization::standardTag("fweights", featureWeightsSS.str(), numTabs) << endl;
72  }

76  void NeuronLearner::load(nor_utils::StreamTokenizer& st)
77  {
78  // Calling the super-class method
79  AbstainableLearner::load(st);
80 
81  _featuresWeight.resize(0);
82 
83  // load feature weights
84  locale tmpLocale = locale(locale(), new nor_utils::white_spaces(", "));
85  string featureWeightsString = UnSerialization::seekAndParseEnclosedValue<string>(st, "fweights");
86 
87  stringstream featureWeightsSS(featureWeightsString);
88  featureWeightsSS.imbue(tmpLocale);
89 
90  // store the weights
91  while (!featureWeightsSS.eof())
92  {
93  FeatureReal tmpWeight;
94  featureWeightsSS >> tmpWeight;
95 
96  _featuresWeight.push_back( tmpWeight );
97  }

100  }
