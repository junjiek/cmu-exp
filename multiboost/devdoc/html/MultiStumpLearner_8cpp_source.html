75  AlphaReal MultiStumpLearner::run() {
76  const int numClasses = _pTrainingData->getNumClasses();
77  const int numColumns = _pTrainingData->getNumAttributes();
78 
79  // set the size of selected columns
80  _selectedColumnArray.resize( numClasses );
81  fill( _selectedColumnArray.begin(), _selectedColumnArray.end(), -1 );
82 
83  // for storing the class-wise maximal edge
84  vector<AlphaReal> classwiseEdge(numClasses);
85  fill(classwiseEdge.begin(), classwiseEdge.end(), -1.0 );
86 
87  //
88  _v.resize(numClasses);
89  _thresholds.resize(numClasses);
90 
91  // set the smoothing value to avoid numerical problem
92  // when theta=0.
93  setSmoothingVal(1.0 / (AlphaReal) _pTrainingData->getNumExamples() * 0.01);
94 
95  vector<sRates> mu(numClasses); // The class-wise rates. See BaseLearner::sRates for more info.
96  vector<sRates> bestmu(numClasses);
97 
98  vector<AlphaReal> tmpV(numClasses); // The class-wise votes/abstentions
99  vector<FeatureReal> tmpThresholds(numClasses);
100 
101  AlphaReal bestEnergy = numeric_limits<float>::max();
102 
103 
104  StumpAlgorithm<FeatureReal> sAlgo(numClasses);
105  sAlgo.initSearchLoop(_pTrainingData);
106 
107  int numOfDimensions = _maxNumOfDimensions;
108  for (int j = 0; j < numColumns; ++j) {
109  // Tricky way to select numOfDimensions columns randomly out of numColumns
110  int rest = numColumns - j;
111  float r = rand() / static_cast<float> (RAND_MAX);
112 
113  if (static_cast<float> (numOfDimensions) / rest > r) {
114  --numOfDimensions;
115  const pair<vpIterator, vpIterator>
116  dataBeginEnd =
117  static_cast<SortedData*> (_pTrainingData)->getFilteredBeginEnd(j);
118 
119  const vpIterator dataBegin = dataBeginEnd.first;
120  const vpIterator dataEnd = dataBeginEnd.second;
121 
122  sAlgo.findMultiThresholdsWithInit(dataBegin, dataEnd,
123  _pTrainingData, tmpThresholds, &mu, &tmpV);
124 
125  for ( vector<sRates>::iterator itR = mu.begin(); itR != mu.end(); ++itR )
126  {
127  AlphaReal tmpEdgePerClass = ( itR->rPls - itR->rMin );
128 
129  // in each iteration the edge will be maximized, here this is done class-wisely.
130  if ( classwiseEdge[itR->classIdx] < tmpEdgePerClass )
131  {
132  classwiseEdge[itR->classIdx] = tmpEdgePerClass;
133  bestmu[itR->classIdx] = *itR;
134 
135  _v[itR->classIdx] = tmpV[itR->classIdx];
136  _selectedColumnArray[itR->classIdx] = j;
137  _thresholds[itR->classIdx] = tmpThresholds[itR->classIdx];
138  //cout << tmpThresholds[itR->classIdx] << endl;
139  }
140 
141  }
142  }
143  }
144 
145  bestEnergy = getEnergy(bestmu, _alpha, _v);
146 
147  return bestEnergy;
148 
149  }
