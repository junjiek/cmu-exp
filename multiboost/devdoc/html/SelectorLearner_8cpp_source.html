46  AlphaReal SelectorLearner::run()
47  {
48  const int numClasses = _pTrainingData->getNumClasses();
49  const int numColumns = _pTrainingData->getNumAttributes();
50  const int numExamples = _pTrainingData->getNumExamples();
51 
52  // set the smoothing value to avoid numerical problem
53  // when theta=0.
54  setSmoothingVal( 1.0 / (AlphaReal)_pTrainingData->getNumExamples() * 0.01 );
55 
56  vector<sRates> vMu(numClasses); // The class-wise rates. See BaseLearner::sRates for more info.
57  vector<AlphaReal> tmpV(numClasses); // The class-wise votes/abstentions
58 
59  AlphaReal tmpAlpha, tmpEnergy;
60  AlphaReal bestEnergy = numeric_limits<AlphaReal>::max();
61 
62  int numOfDimensions = _maxNumOfDimensions;
63  for (int j = 0; j < numColumns; ++j)
64  {
65  // Tricky way to select numOfDimensions columns randomly out of numColumns
66  int rest = numColumns - j;
67  float r = rand()/static_cast<float>(RAND_MAX);
68 
69  if ( static_cast<float>(numOfDimensions) / rest > r )
70  {
71  --numOfDimensions;
72  /*
73  if (_verbose > 2)
74  cout << " --> trying attribute = "
75  <<_pTrainingData->getAttributeNameMap().getNameFromIdx(j)
76  << endl << flush;
77  */
78  const int numIdxs = _pTrainingData->getEnumMap(j).getNumNames();
79 
80  // Create and initialize the numIdxs x numClasses gamma matrix
81  vector<vector<AlphaReal> > tmpGammasPls(numIdxs);
82  vector<vector<AlphaReal> > tmpGammasMin(numIdxs);
83  for (int io = 0; io < numIdxs; ++io) {
84  vector<AlphaReal> tmpGammaPls(numClasses);
85  vector<AlphaReal> tmpGammaMin(numClasses);
86  fill(tmpGammaPls.begin(), tmpGammaPls.end(), 0.0);
87  fill(tmpGammaMin.begin(), tmpGammaMin.end(), 0.0);
88  tmpGammasPls[io] = tmpGammaPls;
89  tmpGammasMin[io] = tmpGammaMin;
90  }
91 
92  // Compute the elements of the gamma plus and minus matrices
93  AlphaReal entry;
94  for (int i = 0; i < numExamples; ++i) {
95  const vector<Label>& labels = _pTrainingData->getLabels(i);
96  int io = static_cast<int>(_pTrainingData->getValue(i,j));
97  for (int l = 0; l < numClasses; ++l) {
98  entry = labels[l].weight * labels[l].y;
99  if (entry > 0)
100  tmpGammasPls[io][l] += entry;
101  else if (entry < 0)
102  tmpGammasMin[io][l] += -entry;
103  }
104  }
105 
106  // Initialize the u vector to random +-1
107  vector<sRates> uMu(numIdxs); // The idx-wise rates
108  vector<AlphaReal> tmpU(numIdxs);// The idx-wise votes/abstentions
109  vector<AlphaReal> previousTmpU(numIdxs);// The idx-wise votes/abstentions
110 
111  for (int io = 0; io < numIdxs; ++io) {
112  uMu[io].classIdx = io;
113  }
114 
115  for (int io = 0; io < numIdxs; ++io) {
116  // initializing u as it has only one positive element
117  fill( tmpU.begin(), tmpU.end(), -1 );
118  tmpU[io] = +1;
119 
120  vector<sRates> vMu(numClasses); // The label-wise rates
121  for (int l = 0; l < numClasses; ++l)
122  vMu[l].classIdx = l;
123  vector<AlphaReal> tmpV(numClasses); // The label-wise votes/abstentions
124 
125  AlphaReal tmpVal;
126  tmpAlpha = 0.0;
127 
128  //filling out tmpV and vMu
129  for (int l = 0; l < numClasses; ++l) {
130  vMu[l].rPls = vMu[l].rMin = vMu[l].rZero = 0;
131  for (int io = 0; io < numIdxs; ++io) {
132  if (tmpU[io] > 0) {
133  vMu[l].rPls += tmpGammasPls[io][l];
134  vMu[l].rMin += tmpGammasMin[io][l];
135  }
136  else if (tmpU[io] < 0) {
137  vMu[l].rPls += tmpGammasMin[io][l];
138  vMu[l].rMin += tmpGammasPls[io][l];
139  }
140  }
141  if (vMu[l].rPls >= vMu[l].rMin) {
142  tmpV[l] = +1;
143  }
144  else {
145  tmpV[l] = -1;
146  tmpVal = vMu[l].rPls;
147  vMu[l].rPls = vMu[l].rMin;
148  vMu[l].rMin = tmpVal;
149  }
150  }
151 
152  tmpEnergy = AbstainableLearner::getEnergy(vMu, tmpAlpha, tmpV);
153 
154  if ( tmpEnergy < bestEnergy && tmpAlpha > 0 ) {
155  _alpha = tmpAlpha;
156  _v = tmpV;
157  //_u = tmpU;
158  _positiveIdxOfArrayU = io;
159  _selectedColumn = j;
160  bestEnergy = tmpEnergy;
161  }
162  }
163  }
164  }
165 
166 
167  if (_selectedColumn>-1)
168  {
169  _id = _pTrainingData->getAttributeNameMap().getNameFromIdx(_selectedColumn);
170  return bestEnergy;
171  } else {
172  return bestEnergy = numeric_limits<float>::signaling_NaN();
173  }
174  }
