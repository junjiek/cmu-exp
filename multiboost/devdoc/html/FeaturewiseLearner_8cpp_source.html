130  AlphaReal FeaturewiseLearner::run( vector<int>& colIndices )
131  {
132  AlphaReal bestEnergy = -numeric_limits<AlphaReal>::max();
133  FeaturewiseLearner* bestClassifier = NULL;
134 
135  // for each fearture given in colIndices run the learner
136  for( vector<int>::iterator it = colIndices.begin(); it != colIndices.end(); ++it )
137  {
138  AlphaReal tmpEnergy = this->run(*it);
139  if ( tmpEnergy > bestEnergy )
140  {
141  // keep the best classifier
142  if (bestClassifier) delete bestClassifier;
143  bestClassifier = reinterpret_cast<FeaturewiseLearner*>(copyState());
144  bestEnergy = tmpEnergy;
145  }
146  }
147 
148  // copy the best classifier
149  subCopyState(bestClassifier);
150  delete bestClassifier;
151 
152  return bestEnergy;
153  };
