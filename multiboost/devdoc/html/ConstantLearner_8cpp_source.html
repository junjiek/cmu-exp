49  AlphaReal ConstantLearner::run()
50  {
51 
52  const int numClasses = _pTrainingData->getNumClasses();
53  //const int numColumns = _pTrainingData->getNumColumns();
54 
55  // set the smoothing value to avoid numerical problem
56  // when theta=0.
57  setSmoothingVal( 1.0 / (AlphaReal)_pTrainingData->getNumExamples() * 0.01 );
58 
59  vector<sRates> mu(numClasses); // The class-wise rates. See BaseLearner::sRates for more info.
60  vector<AlphaReal> tmpV(numClasses); // The class-wise votes/abstentions
61 
62  ConstantAlgorithm cAlgo;
63  cAlgo.findConstant(_pTrainingData,&mu,&tmpV);
64 
65  _v = tmpV;
66 
67  return getEnergy(mu, _alpha, _v);
68  }

95  AlphaReal ConstantLearner::finishLearning()
96  {
97  const int numClasses = _pTrainingData->getNumClasses();
98  AlphaReal bestEdge = 0.0;
99 
100  for(int i=0; i<numClasses;++i)
101  {
102  bestEdge += (_v[i]<0)? -_v[i] : _v[i];
103  _v[i] = (_v[i]<0)? 1.0 : -1.0;
104  }
105 
106  return bestEdge;
107  }

109  AlphaReal ConstantLearner::update( int idx )
110  {
111  vector<Label> labels = _pTrainingData->getLabels(idx);
112  vector<Label>::iterator lIt;
113 
114  for ( lIt = labels.begin(); lIt != labels.end(); ++lIt )
115  {
116  _v[lIt->idx] += (lIt->weight * lIt->y);
117  }
118 
119  return 0.0;
120  }
