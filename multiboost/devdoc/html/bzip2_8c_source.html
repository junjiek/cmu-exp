273 /* Divide *n by 10, and return the remainder. */
274 static
275 Int32 uInt64_qrm10 ( UInt64* n )
276 {
277  UInt32 rem, tmp;
278  Int32 i;
279  rem = 0;
280  for (i = 7; i >= 0; i--) {
281  tmp = rem * 256 + n->b[i];
282  n->b[i] = tmp / 10;
283  rem = tmp % 10;
284  }
285  return rem;
286 }

328 {
329  BZFILE* bzf = NULL;
330  UChar ibuf[5000];
331  Int32 nIbuf;
332  UInt32 nbytes_in_lo32, nbytes_in_hi32;
333  UInt32 nbytes_out_lo32, nbytes_out_hi32;
334  Int32 bzerr, bzerr_dummy, ret;
335 
336  SET_BINARY_MODE(stream);
337  SET_BINARY_MODE(zStream);
338 
339  if (ferror(stream)) goto errhandler_io;
340  if (ferror(zStream)) goto errhandler_io;
341 
342  bzf = BZ2_bzWriteOpen ( &bzerr, zStream,
343  blockSize100k, verbosity, workFactor );
344  if (bzerr != BZ_OK) goto errhandler;
345 
346  if (verbosity >= 2) fprintf ( stderr, "\n" );
347 
348  while (True) {
349 
350  if (myfeof(stream)) break;
351  nIbuf = fread ( ibuf, sizeof(UChar), 5000, stream );
352  if (ferror(stream)) goto errhandler_io;
353  if (nIbuf > 0) BZ2_bzWrite ( &bzerr, bzf, (void*)ibuf, nIbuf );
354  if (bzerr != BZ_OK) goto errhandler;
355 
356  }
357 
358  BZ2_bzWriteClose64 ( &bzerr, bzf, 0,
359  &nbytes_in_lo32, &nbytes_in_hi32,
360  &nbytes_out_lo32, &nbytes_out_hi32 );
361  if (bzerr != BZ_OK) goto errhandler;
362 
363  if (ferror(zStream)) goto errhandler_io;
364  ret = fflush ( zStream );
365  if (ret == EOF) goto errhandler_io;
366  if (zStream != stdout) {
367  Int32 fd = fileno ( zStream );
368  if (fd < 0) goto errhandler_io;
369  applySavedFileAttrToOutputFile ( fd );
370  ret = fclose ( zStream );
371  outputHandleJustInCase = NULL;
372  if (ret == EOF) goto errhandler_io;
373  }
374  outputHandleJustInCase = NULL;
375  if (ferror(stream)) goto errhandler_io;
376  ret = fclose ( stream );
377  if (ret == EOF) goto errhandler_io;
378 
379  if (verbosity >= 1) {
380  if (nbytes_in_lo32 == 0 && nbytes_in_hi32 == 0) {
381  fprintf ( stderr, " no data compressed.\n");
382  } else {
383  Char buf_nin[32], buf_nout[32];
384  UInt64 nbytes_in, nbytes_out;
385  double nbytes_in_d, nbytes_out_d;
386  uInt64_from_UInt32s ( &nbytes_in,
387  nbytes_in_lo32, nbytes_in_hi32 );
388  uInt64_from_UInt32s ( &nbytes_out,
389  nbytes_out_lo32, nbytes_out_hi32 );
390  nbytes_in_d = uInt64_to_double ( &nbytes_in );
391  nbytes_out_d = uInt64_to_double ( &nbytes_out );
392  uInt64_toAscii ( buf_nin, &nbytes_in );
393  uInt64_toAscii ( buf_nout, &nbytes_out );
394  fprintf ( stderr, "%6.3f:1, %6.3f bits/byte, "
395  "%5.2f%% saved, %s in, %s out.\n",
396  nbytes_in_d / nbytes_out_d,
397  (8.0 * nbytes_out_d) / nbytes_in_d,
398  100.0 * (1.0 - nbytes_out_d / nbytes_in_d),
399  buf_nin,
400  buf_nout
401  );
402  }
403  }
404 
405  return;
406 
407 errhandler:
408  BZ2_bzWriteClose64 ( &bzerr_dummy, bzf, 1,
409  &nbytes_in_lo32, &nbytes_in_hi32,
410  &nbytes_out_lo32, &nbytes_out_hi32 );
411  switch (bzerr) {

416  case BZ_IO_ERROR:
417  errhandler_io:
418  ioError(); break;
419  default:
420  panic ( "compress:unexpected error" );
421  }

424  /*notreached*/
425 }

431 Bool uncompressStream ( FILE *zStream, FILE *stream )
432 {
433  BZFILE* bzf = NULL;
434  Int32 bzerr, bzerr_dummy, ret, nread, streamNo, i;
435  UChar obuf[5000];
436  UChar unused[BZ_MAX_UNUSED];
437  Int32 nUnused;
438  void* unusedTmpV;
439  UChar* unusedTmp;
440 
441  nUnused = 0;
442  streamNo = 0;
443 
444  SET_BINARY_MODE(stream);
445  SET_BINARY_MODE(zStream);
446 
447  if (ferror(stream)) goto errhandler_io;
448  if (ferror(zStream)) goto errhandler_io;
449 
450  while (True) {
451 
452  bzf = BZ2_bzReadOpen (
453  &bzerr, zStream, verbosity,
454  (int)smallMode, unused, nUnused
455  );
456  if (bzf == NULL || bzerr != BZ_OK) goto errhandler;
457  streamNo++;
458 
459  while (bzerr == BZ_OK) {
460  nread = BZ2_bzRead ( &bzerr, bzf, obuf, 5000 );
461  if (bzerr == BZ_DATA_ERROR_MAGIC) goto trycat;
462  if ((bzerr == BZ_OK || bzerr == BZ_STREAM_END) && nread > 0)
463  fwrite ( obuf, sizeof(UChar), nread, stream );
464  if (ferror(stream)) goto errhandler_io;
465  }
466  if (bzerr != BZ_STREAM_END) goto errhandler;
467 
468  BZ2_bzReadGetUnused ( &bzerr, bzf, &unusedTmpV, &nUnused );
469  if (bzerr != BZ_OK) panic ( "decompress:bzReadGetUnused" );
470 
471  unusedTmp = (UChar*)unusedTmpV;
472  for (i = 0; i < nUnused; i++) unused[i] = unusedTmp[i];
473 
474  BZ2_bzReadClose ( &bzerr, bzf );
475  if (bzerr != BZ_OK) panic ( "decompress:bzReadGetUnused" );
476 
477  if (nUnused == 0 && myfeof(zStream)) break;
478  }
479 
480 closeok:
481  if (ferror(zStream)) goto errhandler_io;
482  if (stream != stdout) {
483  Int32 fd = fileno ( stream );
484  if (fd < 0) goto errhandler_io;
485  applySavedFileAttrToOutputFile ( fd );
486  }
487  ret = fclose ( zStream );
488  if (ret == EOF) goto errhandler_io;
489 
490  if (ferror(stream)) goto errhandler_io;
491  ret = fflush ( stream );
492  if (ret != 0) goto errhandler_io;
493  if (stream != stdout) {
494  ret = fclose ( stream );
495  outputHandleJustInCase = NULL;
496  if (ret == EOF) goto errhandler_io;
497  }
498  outputHandleJustInCase = NULL;
499  if (verbosity >= 2) fprintf ( stderr, "\n " );
500  return True;
501 
502 trycat:
503  if (forceOverwrite) {
504  rewind(zStream);
505  while (True) {
506  if (myfeof(zStream)) break;
507  nread = fread ( obuf, sizeof(UChar), 5000, zStream );
508  if (ferror(zStream)) goto errhandler_io;
509  if (nread > 0) fwrite ( obuf, sizeof(UChar), nread, stream );
510  if (ferror(stream)) goto errhandler_io;
511  }
512  goto closeok;
513  }
514 
515 errhandler:
516  BZ2_bzReadClose ( &bzerr_dummy, bzf );
517  switch (bzerr) {
518  case BZ_CONFIG_ERROR:
519  configError(); break;
520  case BZ_IO_ERROR:
521  errhandler_io:
522  ioError(); break;
523  case BZ_DATA_ERROR:
524  crcError();
525  case BZ_MEM_ERROR:
526  outOfMemory();
527  case BZ_UNEXPECTED_EOF:
528  compressedStreamEOF();
529  case BZ_DATA_ERROR_MAGIC:
530  if (zStream != stdin) fclose(zStream);
531  if (stream != stdout) fclose(stream);
532  if (streamNo == 1) {
533  return False;
534  } else {
535  if (noisy)
536  fprintf ( stderr,
537  "\n%s: %s: trailing garbage after EOF ignored\n",
538  progName, inName );
539  return True;
540  }
541  default:
542  panic ( "decompress:unexpected error" );
543  }
544 
545  panic ( "decompress:end" );
546  return True; /*notreached*/
547 }

707  retVal = remove ( outName );
708  if (retVal != 0)
709  fprintf ( stderr,
710  "%s: WARNING: deletion of output file "
711  "(apparently) failed.\n",
712  progName );
