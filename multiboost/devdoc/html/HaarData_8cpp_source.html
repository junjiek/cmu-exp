65  string regsString = HaarFeature::RegisteredFeatures().getRegString();
66

68  regsString = args.getValue<string>("ftypes", 0);
69 
70  string::const_iterator it;
71  for (it = regsString.begin(); it != regsString.end(); ++it)
72  {
73  char argStr[3];
74  argStr[0] = *it++;
75 
76  if (it != regsString.end())
77  argStr[1] = *it;

79  argStr[1] = '\0';

83  {
84  cerr << "ERROR: Haar feature declaration <" << argStr << "> not found!" << endl;
85  exit(1);
86  }
87 
88  HaarFeature* hf = HaarFeature::RegisteredFeatures().getFeature(argStr)->create();
89  _loadedFeatures.push_back( hf );
90  }
91 
92  string tmpVal = args.getValue<string>("iisize", 0);
93 
94  size_t divPos = tmpVal.find('x');
95  if (divPos == string::npos)
96  {
97  cerr << "ERROR: syntax error for option -iisize!" << endl;
98  exit(1);
99  }
100 
101  istringstream ss(tmpVal);
102  char val[10];
103  ss.getline(val, 10, 'x');
104  _width = static_cast<short>( atoi(val) );
105 
106  ss.getline(val, 10);

268  bool HaarData::checkInput(const string& line, int numColumns)
269  {
270  istringstream ss(line);
271  ss.imbue( locale(locale(), new nor_utils::white_spaces(_pData->getSepChars()) ) );
272 
273  string tmp;
274  bool inputValid = true;
275 
276  if (_hasExampleName)
277  ss >> tmp; // filename
278 
279  if (!_classInLastColumn)
280  ss >> tmp; // class at the beginning
281 
282  for (int j = 0; j < numColumns; ++j)
283  {
284  if ( ss.eof() )
285  {
286  inputValid = false;
287  break;
288  }
289 
290  ss >> tmp;
291 
292  if ( !nor_utils::is_number(tmp) )
293  {
294  inputValid = false;
295  break;
296  }
297  }
298 
299  if (_classInLastColumn)
300  ss >> tmp; // class
301  if ( tmp.empty() )
302  inputValid = false;
303 
304  return inputValid;
305  }
