222  virtual int getChildIndWithMaxBi() {
223  int retVal = 0;
224  double ucb = 0.0;
225  double maxB = numeric_limits< double >::min();
226  for( int i = 0; i < getNumOfChild(); i++ ) {
227  if ( _children[i] != 0 ) ucb = _children[i]->getUpperConfidenceBound();
228  else ucb = ((double) rand() / (double )RAND_MAX ) + sqrt( ( 2 * log( (double) getNi() )) / 1.0 );
229  if ( maxB < ucb ) {
230  retVal = i;
231  maxB = ucb;
232  }
233  }
234  return retVal;
