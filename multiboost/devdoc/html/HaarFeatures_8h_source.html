432  template <typename TDeriv>
433  void _fillHaarData( const vector<Example>& intImages, // input
434  vector< pair<int, FeatureReal> >& haarData ) // output
435  {
436  int i;
437  vector<Example>::const_iterator iiIt;
438  const vector<Example>::const_iterator iiEnd = intImages.end();
439 
440  vector< pair<int, FeatureReal> >::iterator hIt;
441  const nor_utils::Rect& currConfig = getCurrentConfig();
442 
443  for (iiIt = intImages.begin(), hIt = haarData.begin(), i = 0;
444  iiIt != iiEnd; ++iiIt, ++hIt, ++i)
445  {
446  hIt->first = i;
447  hIt->second = static_cast<TDeriv&>(*this).getValue(iiIt->getValues(), currConfig);
448  }
