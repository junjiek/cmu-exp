229  template <typename T>
230  FeatureReal RegressionStumpAlgorithm<T>::findSingleThresholdWithInit
231  (const vpIterator& dataBegin,const vpIterator& dataEnd,
232  InputData* pData, AlphaReal halfTheta, vector<sRates>* pMu, vector<AlphaReal>* pV, vector<AlphaReal>* pV2 )
233  {
234  const int numClasses = pData->getNumClasses();
235 
236  vpIterator currentSplitPos; // the iterator of the currently examined example
237  vpIterator previousSplitPos; // the iterator of the example before the current example
238  vpIterator bestSplitPos; // the iterator of the best split
239  vpIterator bestPreviousSplitPos; // the iterator of the example before the best split
240 
241  // initialize halfEdges to the constant classifier's half edges
242  copy(_constantHalfEdges.begin(), _constantHalfEdges.end(), _halfEdges.begin());
243 
244  AlphaReal currHalfEdge = 0;
245  AlphaReal bestHalfEdge = -numeric_limits<AlphaReal>::max();
246  vector<Label>::const_iterator lIt;
247 
248  // find the best threshold (cutting point)
249  // at the first split we have
250  // first split: x | x x x x x x x x ..
251  // previous -^ ^- current
252  for( currentSplitPos = previousSplitPos = dataBegin, ++currentSplitPos;
253  currentSplitPos != dataEnd;
254  previousSplitPos = currentSplitPos, ++currentSplitPos)
255  {
256  vector<Label>& labels = pData->getLabels(previousSplitPos->first);
257 
258  // recompute halfEdges at the next point
260  for (lIt = labels.begin(); lIt != labels.end(); ++lIt )
261  _halfEdges[ lIt->idx ] -= lIt->weight * lIt->y;
263 
264  // points with the same value of data: to skip because we cannot find a cutting point here!
265  // so we only do the cutting if there is a "hole":
266  if ( previousSplitPos->second != currentSplitPos->second )
267  {
268 
269  currHalfEdge = 0;
270 
272  if ( nor_utils::is_zero(halfTheta) ) { // we save an "if" in the loop, 20% faster
273  for (int l = 0; l < numClasses; ++l) {
274  // flip the class-wise edge if it is negative
275  // but store the flipping bit only at the end (below**)
276  if ( _halfEdges[l] > 0 )
277  currHalfEdge += _halfEdges[l];
278  else
279  currHalfEdge -= _halfEdges[l];
280  }
281  }
282  else {
283  for (int l = 0; l < numClasses; ++l) {
284  // flip the class-wise edge if it is negative
285  // but store the flipping bit only at the end (below**)
286  if ( _halfEdges[l] > halfTheta )
287  currHalfEdge += _halfEdges[l];
288  else if ( _halfEdges[l] < -halfTheta )
289  currHalfEdge -= _halfEdges[l];
290  }
291  }
293 
294  // the current edge is the new maximum
295  if (currHalfEdge > bestHalfEdge)
296  {
297  bestHalfEdge = currHalfEdge;
298  bestSplitPos = currentSplitPos;
299  bestPreviousSplitPos = previousSplitPos;
300 
301  for (int l = 0; l < numClasses; ++l)
302  _bestHalfEdges[l] = _halfEdges[l];
303  }
304  }
305  }
306 
307  // If we found a valid stump in this dimension
308  if (bestHalfEdge > -numeric_limits<AlphaReal>::max())
309  {
310  float threshold = static_cast<FeatureReal>( bestPreviousSplitPos->second +
311  bestSplitPos->second ) / 2;
312 
313  // Fill the mus if present. This could have been done in the threshold loop,
314  // but here is done just once
315  if ( pMu )
316  {
317  for (int l = 0; l < numClasses; ++l)
318  {
319  // **here
320  if (_bestHalfEdges[l] > 0)
321  (*pV)[l] = +1;
322  else
323  (*pV)[l] = -1;
324 
325  (*pMu)[l].classIdx = l;
326 
327  (*pMu)[l].rPls = _halfWeightsPerClass[l] + (*pV)[l] * _bestHalfEdges[l];
328  (*pMu)[l].rMin = _halfWeightsPerClass[l] - (*pV)[l] * _bestHalfEdges[l];
329  (*pMu)[l].rZero = (*pMu)[l].rPls + (*pMu)[l].rMin; // == weightsPerClass[l]
330  }
331  }
332  //cout << 2 * bestHalfEdge << endl << flush;
333  return threshold;
334  }
335  else
336  return numeric_limits<FeatureReal>::signaling_NaN();
337 
338  } // end of findSingleThresholdWithInit

359  void RegressionStumpAlgorithm<T>::findMultiThresholdsWithInit
360  (const vpIterator& dataBegin, const vpIterator& dataEnd,
361  InputData* pData, vector<FeatureReal>& thresholds,
362  vector<sRates>* pMu, vector<AlphaReal>* pV)
363  {
364  const int numClasses = pData->getNumClasses();
365 
366  vpIterator currentSplitPos; // the iterator of the currently examined example
367  vpIterator previousSplitPos; // the iterator of the example before the current example
368 
369  // Initializing halfEdges to the constant classifier's half edges
370  copy(_constantHalfEdges.begin(), _constantHalfEdges.end(), _halfEdges.begin());
371 
372  // Initializing bestHalfEdges, thresholds, and pV to the constant classifier
373  copy(_constantHalfEdges.begin(), _constantHalfEdges.end(), _bestHalfEdges.begin());
374  for (int l = 0; l < numClasses; ++l)
375  thresholds[l] = -numeric_limits<FeatureReal>::max(); // constant cut
376 
377  bool alignAlloc = false;
378  if (pV == NULL)
379  {
380  pV = new vector<AlphaReal>(numClasses);
381  alignAlloc = true;
382  }
383 
384  for (int l = 0; l < numClasses; ++l)
385  {
386  if (_halfEdges[l] > 0) // constant cut
387  (*pV)[l] = 1;
388  else
389  (*pV)[l] = -1;
390  }
391 
392 
393  vector<Label>::const_iterator lIt;
394 
395  // find the best threshold (cutting point)
396  // at the first split we have
397  // first split: x | x x x x x x x x ..
398  // previous -^ ^- current
399  for( currentSplitPos = previousSplitPos = dataBegin, ++currentSplitPos;
400  currentSplitPos != dataEnd;
401  previousSplitPos = currentSplitPos, ++currentSplitPos)
402  {
403  // recompute halfEdges at the next point
404  // this is the bottleneck
405  vector<Label>& labels = pData->getLabels(previousSplitPos->first);
406 
407  for (lIt = labels.begin(); lIt != labels.end(); ++lIt )
408  _halfEdges[ lIt->idx ] -= lIt->weight * lIt->y;
409 
410  // points with the same value of data: to skip because we cannot find a cutting point here!
411  // so we only do the cutting if there is a "hole":
412  if ( previousSplitPos->second != currentSplitPos->second )
413  {
414  for (int l = 0; l < numClasses; ++l)
415  {
416  if (_halfEdges[l] > 0)
417  {
418  // the current edge is the new maximum
419  if (_halfEdges[l] > _bestHalfEdges[l] * (*pV)[l])
420  {
421  (*pV)[l] = 1;
422  _bestHalfEdges[l] = _halfEdges[l];
423  thresholds[l] = static_cast<FeatureReal>( previousSplitPos->second +
424  currentSplitPos->second ) / 2;
425  }
426  }
427  else
428  {
429  // the current edge is the new maximum
430  if (-_halfEdges[l] > _bestHalfEdges[l] * (*pV)[l])
431  {
432  (*pV)[l] = -1;
433  _bestHalfEdges[l] = _halfEdges[l];
434  thresholds[l] = static_cast<FeatureReal>( previousSplitPos->second +
435  currentSplitPos->second ) / 2;
436  }
437  }
438  }
439  }
440  }
441 
442  // Fill the mus if present. This could have been done in the threshold loop,
443  // but here is done just once
444  if ( pMu )
445  {
446  for (int l = 0; l < numClasses; ++l)
447  {
448  (*pMu)[l].classIdx = l;
449 
450  (*pMu)[l].rPls = _halfWeightsPerClass[l] + (*pV)[l] * _bestHalfEdges[l];
451  (*pMu)[l].rMin = _halfWeightsPerClass[l] - (*pV)[l] * _bestHalfEdges[l];
452  (*pMu)[l].rZero = (*pMu)[l].rPls + (*pMu)[l].rMin; // == weightsPerClass[l]
453  }
454  }
455 
456  if (alignAlloc)
457  delete pV;
458 
459  } // end of findMultiThresholdWithInit
