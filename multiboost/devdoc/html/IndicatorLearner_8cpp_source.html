46  AlphaReal IndicatorLearner::run()
47  {
48  const int numClasses = _pTrainingData->getNumClasses();
49  const int numColumns = _pTrainingData->getNumAttributes();
50  const int numExamples = _pTrainingData->getNumExamples();
51 
52  // set the smoothing value to avoid numerical problem
53  // when theta=0.
54  setSmoothingVal( 1.0 / (float)_pTrainingData->getNumExamples() * 0.01 );
55 
56  vector<sRates> vMu(numClasses); // The class-wise rates. See BaseLearner::sRates for more info.
57  vector<AlphaReal> tmpV(numClasses); // The class-wise votes/abstentions
58  vector<AlphaReal> previousTmpV(numClasses); // The class-wise votes/abstentions
59 
60  AlphaReal tmpAlpha,previousTmpAlpha, previousEnergy;
61  AlphaReal bestEnergy = numeric_limits<float>::max();
62 
63  int numOfDimensions = _maxNumOfDimensions;
64  for (int j = 0; j < numColumns; ++j)
65  {
66  // Tricky way to select numOfDimensions columns randomly out of numColumns
67  int rest = numColumns - j;
68  float r = rand()/static_cast<float>(RAND_MAX);
69 
70  if ( static_cast<float>(numOfDimensions) / rest > r )
71  {
72  --numOfDimensions;
73 
74  if (_verbose > 2)
75  cout << " --> trying attribute = "
76  <<_pTrainingData->getAttributeNameMap().getNameFromIdx(j)
77  << endl << flush;
78 
79  const int numIdxs = _pTrainingData->getEnumMap(j).getNumNames();
80 
81  // Create and initialize the numIdxs x numClasses gamma matrix
82  vector<vector<AlphaReal> > tmpGammasPls(numIdxs);
83  vector<vector<AlphaReal> > tmpGammasMin(numIdxs);
84  for (int io = 0; io < numIdxs; ++io) {
85  vector<AlphaReal> tmpGammaPls(numClasses);
86  vector<AlphaReal> tmpGammaMin(numClasses);
87  fill(tmpGammaPls.begin(), tmpGammaPls.end(), 0.0);
88  fill(tmpGammaMin.begin(), tmpGammaMin.end(), 0.0);
89  tmpGammasPls[io] = tmpGammaPls;
90  tmpGammasMin[io] = tmpGammaMin;
91  }
92 
93  // Compute the elements of the gamma plus and minus matrices
94  AlphaReal entry;
95  for (int i = 0; i < numExamples; ++i) {
96  const vector<Label>& labels = _pTrainingData->getLabels(i);
97  int io = static_cast<int>(_pTrainingData->getValue(i,j));
98  for (int l = 0; l < numClasses; ++l) {
99  entry = labels[l].weight * labels[l].y;
100  if (entry > 0)
101  tmpGammasPls[io][l] += entry;
102  else if (entry < 0)
103  tmpGammasMin[io][l] += -entry;
104  }
105  }
106 
107  // Initialize the u vector to random +-1
108  vector<sRates> uMu(numIdxs); // The idx-wise rates
109  vector<AlphaReal> tmpU(numIdxs);// The idx-wise votes/abstentions
110  vector<AlphaReal> previousTmpU(numIdxs);// The idx-wise votes/abstentions
111  for (int io = 0; io < numIdxs; ++io) {
112  uMu[io].classIdx = io;
113  if ( rand()/static_cast<AlphaReal>(RAND_MAX) > 0.5 )
114  tmpU[io] = +1;
115  else
116  tmpU[io] = -1;
117  }
118 
119  vector<sRates> vMu(numClasses); // The label-wise rates
120  for (int l = 0; l < numClasses; ++l)
121  vMu[l].classIdx = l;
122  vector<AlphaReal> tmpV(numClasses); // The label-wise votes/abstentions
123 
124  AlphaReal tmpEnergy = numeric_limits<AlphaReal>::max();
125  AlphaReal tmpVal;
126  tmpAlpha = 0.0;
127 
128  while (1) {
129  previousEnergy = tmpEnergy;
130  previousTmpV = tmpV;
131  previousTmpAlpha = tmpAlpha;
132 
133  //filling out tmpV and vMu
134  for (int l = 0; l < numClasses; ++l) {
135  vMu[l].rPls = vMu[l].rMin = vMu[l].rZero = 0;
136  for (int io = 0; io < numIdxs; ++io) {
137  if (tmpU[io] > 0) {
138  vMu[l].rPls += tmpGammasPls[io][l];
139  vMu[l].rMin += tmpGammasMin[io][l];
140  }
141  else if (tmpU[io] < 0) {
142  vMu[l].rPls += tmpGammasMin[io][l];
143  vMu[l].rMin += tmpGammasPls[io][l];
144  }
145  }
146  if (vMu[l].rPls >= vMu[l].rMin) {
147  tmpV[l] = +1;
148  }
149  else {
150  tmpV[l] = -1;
151  tmpVal = vMu[l].rPls;
152  vMu[l].rPls = vMu[l].rMin;
153  vMu[l].rMin = tmpVal;
154  }
155  }
156 
157  tmpEnergy = AbstainableLearner::getEnergy(vMu, tmpAlpha, tmpV);
158 
159  if (_verbose > 2)
160  cout << " --> energy V = " << tmpEnergy << "\talpha = " << tmpAlpha << endl << flush;
161 
162  if (tmpEnergy >= previousEnergy) {
163  tmpV = previousTmpV;
164  break;
165  }
166 
167  previousEnergy = tmpEnergy;
168  previousTmpU = tmpU;
169  previousTmpAlpha = tmpAlpha;
170 
171  //filling out tmpU and uMu
172  for (int io = 0; io < numIdxs; ++io) {
173  uMu[io].rPls = uMu[io].rMin = uMu[io].rZero = 0;
174  for (int l = 0; l < numClasses; ++l) {
175  if (tmpV[l] > 0) {
176  uMu[io].rPls += tmpGammasPls[io][l];
177  uMu[io].rMin += tmpGammasMin[io][l];
178  }
179  else if (tmpV[l] < 0) {
180  uMu[io].rPls += tmpGammasMin[io][l];
181  uMu[io].rMin += tmpGammasPls[io][l];
182  }
183  }
184  if (uMu[io].rPls >= uMu[io].rMin) {
185  tmpU[io] = +1;
186  }
187  else {
188  tmpU[io] = -1;
189  tmpVal = uMu[io].rPls;
190  uMu[io].rPls = uMu[io].rMin;
191  uMu[io].rMin = tmpVal;
192  }
193  }
194 
195  tmpEnergy = AbstainableLearner::getEnergy(uMu, tmpAlpha, tmpU);
196 
197  if (_verbose > 2)
198  cout << " --> energy U = " << tmpEnergy << "\talpha = " << tmpAlpha << endl << flush;
199 
200  if (tmpEnergy >= previousEnergy) {
201  tmpU = previousTmpU;
202  break;
203  }
204  }
205 
206  if ( previousEnergy < bestEnergy && previousTmpAlpha > 0 ) {
207  _alpha = previousTmpAlpha;
208  _v = tmpV;
209  _u = tmpU;
210  _selectedColumn = j;
211  bestEnergy = previousEnergy;
212  }
213  }
214  }
215 
216  if (_selectedColumn>-1)
217  {
218  _id = _pTrainingData->getAttributeNameMap().getNameFromIdx(_selectedColumn);
219  return bestEnergy;
220  } else {
221  return bestEnergy = numeric_limits<float>::signaling_NaN();
222  }
