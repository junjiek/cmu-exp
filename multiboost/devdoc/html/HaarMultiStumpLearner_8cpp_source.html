84  AlphaReal HaarMultiStumpLearner::run()
85  {
86  const int numClasses = _pTrainingData->getNumClasses();
87 
88  // set the smoothing value to avoid numerical problem
89  // when theta=0.
90  setSmoothingVal( 1.0 / (AlphaReal)_pTrainingData->getNumExamples() * 0.01 );
91 
92  vector<sRates> mu(numClasses); // The class-wise rates. See BaseLearner::sRates for more info.
93  vector<AlphaReal> tmpV(numClasses); // The class-wise votes/abstentions
94 
95  vector<FeatureReal> tmpThresholds(numClasses);
96  AlphaReal tmpAlpha;
97 
98  AlphaReal bestEnergy = numeric_limits<AlphaReal>::max();
99  AlphaReal tmpEnergy;
100 
101  HaarData* pHaarData = static_cast<HaarData*>(_pTrainingData);
102 
103  // get the whole data matrix
104  // const vector<int*>& intImages = pHaarData->getIntImageVector();
105 
106  // The data matrix transformed into the feature's space
107  vector< pair<int, FeatureReal> > processedHaarData(_pTrainingData->getNumExamples());
108 
109  // I need to prepare both type of sampling
110  int numConf; // for ST_NUM
111  time_t startTime, currentTime; // for ST_TIME
112 
113  long numProcessed;
114  bool quitConfiguration;
115 
116  StumpAlgorithm<FeatureReal> sAlgo(numClasses);
117  sAlgo.initSearchLoop(_pTrainingData);
118 
119  // The declared features types
120  vector<HaarFeature*>& loadedFeatures = pHaarData->getLoadedFeatures();
121 
122  // for every feature type
123  vector<HaarFeature*>::iterator ftIt;
124  for (ftIt = loadedFeatures.begin(); ftIt != loadedFeatures.end(); ++ftIt)
125  {
126  // just for readability
127  HaarFeature* pCurrFeature = *ftIt;
128  if (_samplingType != ST_NO_SAMPLING)
129  pCurrFeature->setAccessType(AT_RANDOM_SAMPLING);
130 
131  // Reset the iterator on the configurations. For random sampling
132  // this shuffles the configurations.
133  pCurrFeature->resetConfigIterator();
134  quitConfiguration = false;
135  numProcessed = 0;
136 
137  numConf = 0;
138  time( &startTime );
139 
140  if (_verbose > 1)
141  cout << "Learning type " << pCurrFeature->getName() << ".." << flush;
142 
143  // While there is a configuration available
144  while ( pCurrFeature->hasConfigs() )
145  {
146  // transform the data from intImages to the feature's space
147  pCurrFeature->fillHaarData(_pTrainingData->getExamples(), processedHaarData);
148  // sort the examples in the new space by their coordinate
149  sort( processedHaarData.begin(), processedHaarData.end(),
150  nor_utils::comparePair<2, int, float, less<float> >() );
151 
152  // find the optimal threshold
153  sAlgo.findMultiThresholdsWithInit(processedHaarData.begin(), processedHaarData.end(),
154  _pTrainingData, tmpThresholds, &mu, &tmpV);
155 
156  tmpEnergy = getEnergy(mu, tmpAlpha, tmpV);
157  ++numProcessed;
158 
159  if (tmpEnergy < bestEnergy)
160  {
161  // Store it in the current weak hypothesis.
162  // note: I don't really like having so many temp variables
163  // but the alternative would be a structure, which would need
164  // to be inheritable to make things more consistent. But this would
165  // make it less flexible. Therefore, I am still undecided. This
166  // might change!
167  _alpha = tmpAlpha;
168  _v = tmpV;
169 
170  // I need to save the configuration because it changes within the object
171  _selectedConfig = pCurrFeature->getCurrentConfig();
172  // I save the object because it contains the informations about the type,
173  // the name, etc..
174  _pSelectedFeature = pCurrFeature;
175  _thresholds = tmpThresholds;
176 
177  bestEnergy = tmpEnergy;
178  }
179 
180  // Move to the next configuration
181  pCurrFeature->moveToNextConfig();
182 
183  // check stopping criterion for random configurations
184  switch (_samplingType)
185  {
186  case ST_NUM:
187  ++numConf;
188  if (numConf >= _samplingVal)
189  quitConfiguration = true;
190  break;
191  case ST_TIME:
192  {
193  time( &currentTime );
194  float diff = difftime(currentTime, startTime); // difftime is in seconds
195  if (diff >= _samplingVal)
196  quitConfiguration = true;
197  }
198  break;
199  case ST_NO_SAMPLING:
200  perror("ERROR: st no sampling... not sure what this means");
201 
202  break;
203 
204  } // end switch
205 
206  if (quitConfiguration)
207  break;
208 
209  } // end while
210 
211  if (_verbose > 1)
212  {
213  time( &currentTime );
214  float diff = difftime(currentTime, startTime); // difftime is in seconds
215 
216  cout << "done! "
217  << "(processed: " << numProcessed
218  << " - elapsed: " << diff << " sec)"
219  << endl;
220  }
221 
222  }
223 
224  if (!_pSelectedFeature)
225  {
226  cerr << "ERROR: No Haar Feature found. Something must be wrong!" << endl;
227  exit(1);
228  }
229  else
230  {
231  if (_verbose > 1)
232  cout << "Selected type: " << _pSelectedFeature->getName() << endl;
233  }
234 
235  return bestEnergy;
236  }
