54  void UCBVHaarSingleStumpLearner::init() {
55  UCBVHaarSingleStumpLearner::_featuresData.clear();
56  UCBVHaarSingleStumpLearner::_numOfCalling = 1;
57 
58  //choose randomly the feature for the UCBV
59  HaarData* pHaarData = static_cast<HaarData*>(_pTrainingData);
60  vector<HaarFeature*>& loadedFeatures = pHaarData->getLoadedFeatures();
61 
62  int numConf = 0;
63  bool quitConfiguration;
64  long numProcessed;
65  FeatureDataUCBV tmpFeatureData;
66  int confIdx = 0;
67 
68  tmpFeatureData.T = 0;
69  tmpFeatureData.X.clear();
70 
71  vector<HaarFeature*>::iterator ftIt;
72  for (ftIt = loadedFeatures.begin(), confIdx = 0; ftIt != loadedFeatures.end(); ++ftIt, confIdx++)
73  {
74  // just for readability
75  HaarFeature* pCurrFeature = *ftIt;
76  if (_samplingType != ST_NO_SAMPLING)
77  pCurrFeature->setAccessType(AT_RANDOM_SAMPLING);
78 
79  // Reset the iterator on the configurations. For random sampling
80  // this clear the visited list
81  pCurrFeature->resetConfigIterator();
82  quitConfiguration = false;
83  numProcessed = 0;
84 
85  numConf = 0;
86 
87  if (_verbose > 1)
88  cout << "Learning type " << pCurrFeature->getName() << ".." << flush;
89 
90  while ( pCurrFeature->hasConfigs() )
91  {
92  // Move to the next configuration
93  pCurrFeature->moveToNextConfig();
94 
95  int key = ( 10 * pCurrFeature->getLoadedConfigIndex() ) + confIdx;
96  //cout << key << "\t" << pCurrFeature->getType() << endl;
97  //cout << (int) (key / 10) << "\t" << (key % 10) << endl;
98 
99  UCBVHaarSingleStumpLearner::_featuresData[key] = tmpFeatureData;
100 
101  // check stopping criterion for random configurations
102  switch (_samplingType)
103  {
104  case ST_NUM:
105  ++numConf;
106  if (numConf >= _samplingVal)
107  quitConfiguration = true;
108  break;
109 
110  case ST_NO_SAMPLING:
111  perror("ERROR: What? No sampling??");
112  break;
113 
114  } // end switch
115 
116  if (quitConfiguration)
117  break;
118 
119  } // end while
120  }
121 
122  cout << "The number of the randomly chosen features:\t" << UCBVHaarSingleStumpLearner::_featuresData.size() << endl;
123  }

167  AlphaReal UCBVHaarSingleStumpLearner::getBValue( int key )
168  {
169  AlphaReal retval = numeric_limits<AlphaReal>::max();
170  map<int,FeatureDataUCBV>::const_iterator itUCBV = UCBVHaarSingleStumpLearner::_featuresData.find( key );
171 
172  if ( itUCBV != UCBVHaarSingleStumpLearner::_featuresData.end() )
173  {
174  int s = (int)itUCBV->second.X.size();
175 
176  if ( s == 0 ) return retval;
177 
178  AlphaReal epsilonT = (AlphaReal) getTthSeriesElement( UCBVHaarSingleStumpLearner::_numOfCalling );
179  retval = (3.0 * epsilonT ) / s;
180  //claculate the average reward
181  AlphaReal avgX = 0.0;
182  for( int i = 0; i < s; i++ )
183  {
184  avgX += itUCBV->second.X[i];
185  }
186  avgX /= s;
187 
188  //claculate the variance of rewards
189  AlphaReal varX = 0.0;
190  for( int i = 0; i < s; i++ )
191  {
192  varX += ( itUCBV->second.X[i] - avgX ) * ( itUCBV->second.X[i] - avgX );
193  }
194  varX /= s;
195 
196  retval += ( avgX + sqrt( ( 2.0 * varX * epsilonT ) / ( s ) ) );
197  }
198 
199  return retval;
200  }

204  AlphaReal UCBVHaarSingleStumpLearner::run()
205  {
206  if ( UCBVHaarSingleStumpLearner::_numOfCalling == 0 ) {
207  init();
208  }
209 
210  UCBVHaarSingleStumpLearner::_numOfCalling++;
211  //cout << "Num of iter:\t" << UCBVHaarSingleStumpLearner::_numOfCalling << " " << this->getTthSeriesElement( UCBVHaarSingleStumpLearner::_numOfCalling ) << flush << endl;
212  const int numClasses = _pTrainingData->getNumClasses();
213 
214  // set the smoothing value to avoid numerical problem
215  // when theta=0.
216  setSmoothingVal( 1.0 / (AlphaReal)_pTrainingData->getNumExamples() * 0.01 );
217 
218  vector<sRates> mu(numClasses); // The class-wise rates. See BaseLearner::sRates for more info.
219  vector<AlphaReal> tmpV(numClasses); // The class-wise votes/abstentions
220 
221  AlphaReal tmpThreshold;
222  AlphaReal tmpAlpha;
223 
224  AlphaReal bestEnergy = numeric_limits<AlphaReal>::max();
225  AlphaReal tmpEnergy;
226 
227  HaarData* pHaarData = static_cast<HaarData*>(_pTrainingData);
228 
229  // get the whole data matrix
230  //const vector<int*>& intImages = pHaarData->getIntImageVector();
231 
232  // The data matrix transformed into the feature's space
233  vector< pair<int, FeatureReal> > processedHaarData(_pTrainingData->getNumExamples());
234 
235  // I need to prepare both type of sampling
236 
237  StumpAlgorithm<FeatureReal> sAlgo(numClasses);
238  sAlgo.initSearchLoop(_pTrainingData);
239 
240  AlphaReal halfTheta;
241  if ( _abstention == ABST_REAL || _abstention == ABST_CLASSWISE )
242  halfTheta = _theta/2.0;
243  else
244  halfTheta = 0;
245 
246  // The declared features types
247  vector<HaarFeature*>& loadedFeatures = pHaarData->getLoadedFeatures();
248 
249  // for every feature type
250  vector<HaarFeature*>::iterator ftIt;
251  //vector<HaarFeature*>::iterator maxftIt;
252 
253  vector<AlphaReal> maxV( loadedFeatures.size() );
254  vector<int> maxKey( loadedFeatures.size() );
255  vector<int> maxNum( loadedFeatures.size() );
256 
257  //claculate the Bk,s,t of the randomly chosen features
258  int key = getKeyOfMaximalElement();
259  int featureIdx = (int) (key / 10);
260  int featureType = (key % 10);
261 
262  //for (i = 0, ftIt = loadedFeatures.begin(); ftIt != loadedFeatures.end(); i++ ++ftIt)
263  //*ftIt = loadedFeatures[ featureType ];
264 
265  // just for readability
266  //HaarFeature* pCurrFeature = *ftIt;
267  HaarFeature* pCurrFeature = loadedFeatures[ featureType ];
268  if (_samplingType != ST_NO_SAMPLING)
269  pCurrFeature->setAccessType(AT_RANDOM_SAMPLING);
270 
271  // Reset the iterator on the configurations. For random sampling
272  // this clear the visited list
273  pCurrFeature->loadConfigByNum( featureIdx );
274 
275 
276  if (_verbose > 1)
277  cout << "Learning type " << pCurrFeature->getName() << ".." << flush;
278 
279  // transform the data from intImages to the feature's space
280  pCurrFeature->fillHaarData( _pTrainingData->getExamples(), processedHaarData );
281  //pCurrFeature->fillHaarData(intImages, processedHaarData);
282 
283  // sort the examples in the new space by their coordinate
284  sort( processedHaarData.begin(), processedHaarData.end(),
285  nor_utils::comparePair<2, int, float, less<float> >() );
286 
287  // find the optimal threshold
288  tmpThreshold = sAlgo.findSingleThresholdWithInit(processedHaarData.begin(),
289  processedHaarData.end(),
290  _pTrainingData, halfTheta, &mu, &tmpV);
291 
292  tmpEnergy = getEnergy(mu, tmpAlpha, tmpV);
293 
294 
295  // Store it in the current weak hypothesis.
296  // note: I don't really like having so many temp variables
297  // but the alternative would be a structure, which would need
298  // to be inheritable to make things more consistent. But this would
299  // make it less flexible. Therefore, I am still undecided. This
300  // might change!
301  _alpha = tmpAlpha;
302  _v = tmpV;
303 
304  // I need to save the configuration because it changes within the object
305  _selectedConfig = pCurrFeature->getCurrentConfig();
306  // I save the object because it contains the informations about the type,
307  // the name, etc..
308  _pSelectedFeature = pCurrFeature;
309  _threshold = tmpThreshold;
310 
311  bestEnergy = tmpEnergy;
312 
313  AlphaReal edge = 0.0;
314  for( vector<sRates>::iterator itR = mu.begin(); itR != mu.end(); itR++ ) edge += ( itR->rPls - itR->rMin );
315  //need to set the X value
316  updateKeys( key, edge * edge );
317 
318  if (!_pSelectedFeature)
319  {
320  cerr << "ERROR: No Haar Feature found. Something must be wrong!" << endl;
321  exit(1);
322  }
323  else
324  {
325  if (_verbose > 1)
326  cout << "Selected type: " << _pSelectedFeature->getName() << endl;
327  }
328 
329  return bestEnergy;
330  }

332  int UCBVHaarSingleStumpLearner::getKeyOfMaximalElement()
333  {
334  int key;
335  float maxVal = numeric_limits<float>::min();
336  map<int,FeatureDataUCBV>::iterator itUCBV;
337 
338  for( itUCBV = UCBVHaarSingleStumpLearner::_featuresData.begin(); itUCBV != _featuresData.end(); itUCBV++ )
339  {
340  float val = getBValue( itUCBV->first );
341  if ( val > maxVal )
342  {
343  maxVal = val;
344  key = itUCBV->first;
345 
346  if ( maxVal == numeric_limits<float>::max() ) break;
347  }
348  }
349  return key;
350  }

353  void UCBVHaarSingleStumpLearner::updateKeys( int key, float val )
354  {
355  map<int,FeatureDataUCBV>::const_iterator itUCBV = UCBVHaarSingleStumpLearner::_featuresData.find( key );
356  FeatureDataUCBV tmp;
357 
358  if ( itUCBV == UCBVHaarSingleStumpLearner::_featuresData.end() )
359  {
360  tmp.T = 1;
361  tmp.X.push_back( val );
362  UCBVHaarSingleStumpLearner::_featuresData[ key ] = tmp;
363  } else {
364  tmp = itUCBV->second;
365  tmp.T++;
366  tmp.X.push_back( val );
367  UCBVHaarSingleStumpLearner::_featuresData[ key ] = tmp;
368  }
