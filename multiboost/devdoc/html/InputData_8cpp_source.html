85  int InputData::loadIndexSet( set< int > ind ) {
86  int i = 0;
87  //upload the indirection
88  for (int j = 0; j < this->_rawIndices.size(); j++) this->_rawIndices[j] = -1;
89 
90  map<int, int> tmpPointsPerClass;
91 
92  for( set< int >::iterator it = ind.begin(); it != ind.end(); it++ ) {
93  this->_indirectIndices[i] = *it;
94  this->_rawIndices[*it] = i;
95 
96  i++;
97 
98  const vector<Label>& labels = _pData->getLabels( *it );
99  vector<Label>::const_iterator lIt;
100 
101  for (lIt = labels.begin(); lIt != labels.end(); ++lIt )
102  {
103  switch ( this->_pData->getLabelRep() ) {
104  case LR_DENSE:
105  if ( lIt->y > 0 )
106  tmpPointsPerClass[lIt->idx]++;
107  break;
108  case LR_SPARSE:
109  if ( lIt->y > 0 )
110  tmpPointsPerClass[lIt->idx]++;
111  break;
112  }
113  }
114  }
115 
116  _nExamplesPerClass.clear();
117  for (int l = 0; l < this->_pData->getNumClasses(); ++l)
118  _nExamplesPerClass.push_back( tmpPointsPerClass[l] );
119 
120 
121  this->_numExamples = (int)ind.size();
122 
123  _subsetAlreadyComputed = false;
124 
125  return 0;
126  }
