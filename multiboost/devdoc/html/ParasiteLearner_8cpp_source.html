94  AlphaReal ParasiteLearner::run()
95  {
96  if (_baseLearners.size() == 0) {
97  // load the base learners
98  if (_verbose >= 2)
99  cout << "loading " << _nameBaseLearnerFile << ".." << flush;
100  UnSerialization us;
101  us.loadHypotheses( _nameBaseLearnerFile, _baseLearners, _pTrainingData, _verbose);
102  if (_verbose >= 2)
103  cout << "finished " << endl << flush;
104  }
105 
106  if ( _numBaseLearners == -1 || _numBaseLearners > _baseLearners.size())
107  _numBaseLearners = _baseLearners.size();
108 
109 
110  const int numClasses = _pTrainingData->getNumClasses();
111  const int numExamples = _pTrainingData->getNumExamples();
112  float tmpAlpha;
113  float bestE = numeric_limits<float>::max();
114  float sumGamma, bestSumGamma = -numeric_limits<float>::max();
115  float tmpE, gamma;
116  float eps_min,eps_pls;
117  int tmpSignOfAlpha;
118 
119  // This is the bottleneck, squeeze out every microsecond
120  if (_closed) {
121  bestSumGamma = 0;
122  if ( nor_utils::is_zero(_theta) ) {
123  for (int j = 0; j < _numBaseLearners; ++j) {
124  sumGamma = 0;
125  for (int i = 0; i < numExamples; ++i) {
126  vector<Label> labels = _pTrainingData->getLabels(i);
127  for (int l = 0; l < numClasses; ++l)
128  sumGamma += labels[l].weight *
129  _baseLearners[j]->classify(_pTrainingData,i,l) * labels[l].y;
130  }
131  if (fabs(sumGamma) > fabs(bestSumGamma)) {
132  _selectedIdx = j;
133  bestSumGamma = sumGamma;
134  }
135  }
136  eps_pls = eps_min = 0;
137  for (int i = 0; i < numExamples; ++i) {
138  vector<Label> labels = _pTrainingData->getLabels(i);
139  for (int l = 0; l < numClasses; ++l) {
140  gamma = _baseLearners[_selectedIdx]->classify(_pTrainingData,i,l) *
141  labels[l].y;
142  if ( gamma > 0 )
143  eps_pls += labels[l].weight;
144  else if ( gamma < 0 )
145  eps_min += labels[l].weight;
146  }
147  }
148  if (eps_min > eps_pls) {
149  float tmpSwap = eps_min;
150  eps_min = eps_pls;
151  eps_pls = tmpSwap;
152  _signOfAlpha = -1;
153  }
154  _alpha = getAlpha(eps_min, eps_pls);
155  bestE = BaseLearner::getEnergy( eps_min, eps_pls );
156  }
157  else {
158  for (int j = 0; j < _numBaseLearners; ++j) {
159  eps_pls = eps_min = 0;
160  for (int i = 0; i < numExamples; ++i) {
161  vector<Label> labels = _pTrainingData->getLabels(i);
162  for (int l = 0; l < numClasses; ++l) {
163  gamma = _baseLearners[j]->classify(_pTrainingData,i,l) * labels[l].y;
164  if ( gamma > 0 )
165  eps_pls += labels[l].weight;
166  else if ( gamma < 0 )
167  eps_min += labels[l].weight;
168  }
169  }
170  if (eps_min > eps_pls) {
171  float tmpSwap = eps_min;
172  eps_min = eps_pls;
173  eps_pls = tmpSwap;
174  tmpSignOfAlpha = -1;
175  }
176  else
177  tmpSignOfAlpha = 1;
178  tmpAlpha = getAlpha(eps_min, eps_pls, _theta);
179  tmpE = BaseLearner::getEnergy( eps_min, eps_pls, tmpAlpha, _theta );
180  if (tmpE < bestE && eps_pls > eps_min + _theta) {
181  _alpha = tmpAlpha;
182  _selectedIdx = j;
183  _signOfAlpha = tmpSignOfAlpha;
184  bestE = tmpE;
185  }
186  }
187  }
188  }
189  else {
190  if ( nor_utils::is_zero(_theta) ) {
191  for (int j = 0; j < _numBaseLearners; ++j) {
192  sumGamma = 0;
193  for (int i = 0; i < numExamples; ++i) {
194  vector<Label> labels = _pTrainingData->getLabels(i);
195  for (int l = 0; l < numClasses; ++l)
196  sumGamma += labels[l].weight *
197  _baseLearners[j]->classify(_pTrainingData,i,l) * labels[l].y;
198  }
199  if (sumGamma > bestSumGamma) {
200  _selectedIdx = j;
201  bestSumGamma = sumGamma;
202  }
203  }
204  eps_pls = eps_min = 0;
205  for (int i = 0; i < numExamples; ++i) {
206  vector<Label> labels = _pTrainingData->getLabels(i);
207  for (int l = 0; l < numClasses; ++l) {
208  gamma = _baseLearners[_selectedIdx]->classify(_pTrainingData,i,l) *
209  labels[l].y;
210  if ( gamma > 0 )
211  eps_pls += labels[l].weight;
212  else if ( gamma < 0 )
213  eps_min += labels[l].weight;
214  }
215  }
216  _alpha = getAlpha(eps_min, eps_pls);
217  bestE = BaseLearner::getEnergy( eps_min, eps_pls );
218  }
219  else {
220  for (int j = 0; j < _numBaseLearners; ++j) {
221  eps_pls = eps_min = 0;
222  for (int i = 0; i < numExamples; ++i) {
223  vector<Label> labels = _pTrainingData->getLabels(i);
224  for (int l = 0; l < numClasses; ++l) {
225  gamma = _baseLearners[j]->classify(_pTrainingData,i,l) * labels[l].y;
226  if ( gamma > 0 )
227  eps_pls += labels[l].weight;
228  else if ( gamma < 0 )
229  eps_min += labels[l].weight;
230  }
231  }
232  tmpAlpha = getAlpha(eps_min, eps_pls, _theta);
233  tmpE = BaseLearner::getEnergy( eps_min, eps_pls, tmpAlpha, _theta );
234  if (tmpE < bestE && eps_pls > eps_min + _theta) {
235  _alpha = tmpAlpha;
236  _selectedIdx = j;
237  bestE = tmpE;
238  }
239  //cout << j << ": e- = " << eps_min << "\t e+ = " << eps_pls << "\t edge = " << (eps_pls - eps_min) << "\t energy = " << tmpE << "\t energy* = " << bestE << "\t alpha = " << tmpAlpha << endl << flush;
240  }
241  }
242  }
243  return bestE;
244 
245  }

263  void ParasiteLearner::load(nor_utils::StreamTokenizer& st)
264  {
265  // cout << "Sorry, you can't load a ParasiteLearner" << endl << flush;
266  // exit(1);

271  _nameBaseLearnerFile = UnSerialization::seekAndParseEnclosedValue<string>(st, "poolfile");
272  _selectedIdx = UnSerialization::seekAndParseEnclosedValue<int>(st, "learneridx");
273 
274  if (_baseLearners.size() == 0) {
275  // load the base learners
276  if (_verbose >= 2)
277  cout << "loading " << _nameBaseLearnerFile << ".." << flush;
278  UnSerialization us;
279  us.loadHypotheses( _nameBaseLearnerFile, _baseLearners, _pTrainingData, _verbose);
280  if (_verbose >= 2)
281  cout << "finished " << endl << flush;
282  }
