154  void AdaBoostMHLearner::run(const nor_utils::Args& args)
155  {
156  // load the arguments
157  this->getArgs(args);
158 
159  // get the registered weak learner (type from name)
160  BaseLearner* pWeakHypothesisSource =
161  BaseLearner::RegisteredLearners().getLearner(_baseLearnerName);
162  // initialize learning options; normally it's done in the strong loop
163  // also, here we do it for Product learners, so input data can be created
164  pWeakHypothesisSource->initLearningOptions(args);
165 
166  BaseLearner* pConstantWeakHypothesisSource =
167  BaseLearner::RegisteredLearners().getLearner("ConstantLearner");
168 
169  // get the training input data, and load it
170 
171  InputData* pTrainingData = pWeakHypothesisSource->createInputData();
172  pTrainingData->initOptions(args);
173  pTrainingData->load(_trainFileName, IT_TRAIN, _verbose);
174 
175  // get the testing input data, and load it
176  InputData* pTestData = NULL;
177  if ( !_testFileName.empty() )
178  {
179  pTestData = pWeakHypothesisSource->createInputData();
180  pTestData->initOptions(args);
181  pTestData->load(_testFileName, IT_TEST, _verbose);
182  }
183 
184  // The output information object
185  OutputInfo* pOutInfo = NULL;
186  if ( !_outputInfoFile.empty() )
187  {
188  pOutInfo = new OutputInfo(args);
189  if (_earlyStopping && pOutInfo->getOutputInfoObject(_earlyStoppingOutputColumn) == NULL)
190  {
191  if (_verbose >= 2)
192  {
193  cout << _earlyStoppingOutputColumn << " not declared as outputinfo but used for earlystopping, automatically added." << endl;
194  }
195  pOutInfo->setOutputList(_earlyStoppingOutputColumn);
196  }
197  pOutInfo->initialize(pTrainingData);
198 
199  if (pTestData)
200  pOutInfo->initialize(pTestData);
201 
202  pOutInfo->outputHeader(pTrainingData->getClassMap());
203 
204 
205  if ( ! args.hasArgument("resume") )
206  {
207  // Baseline: constant classifier - goes into 0th iteration
208 
209  BaseLearner* pConstantWeakHypothesis = pConstantWeakHypothesisSource->create() ;
210  pConstantWeakHypothesis->initLearningOptions(args);
211  pConstantWeakHypothesis->setTrainingData(pTrainingData);
212  pConstantWeakHypothesis->run();
213 
214  pOutInfo->outputIteration(-1);
215  pOutInfo->outputCustom(pTrainingData, pConstantWeakHypothesis);
216  if (pTestData != NULL)
217  {
218  pOutInfo->separator();
219  pOutInfo->outputCustom(pTestData, pConstantWeakHypothesis);
220  }
221  pOutInfo->outputCurrentTime();
222  pOutInfo->endLine();
223 
224  pOutInfo->initialize(pTrainingData);
225  if (pTestData)
226  pOutInfo->initialize(pTestData);
227  }
228  }
229 
230  //cout << "Before serialization" << endl;
231  // reload the previously found weak learners if -resume is set.
232  // otherwise just return 0
233  int startingIteration = resumeWeakLearners(pTrainingData);
234 
235 
236  Serialization ss(_shypFileName, _isShypCompressed );
237  ss.writeHeader(_baseLearnerName); // this must go after resumeProcess has been called
238 
239  // Initialization for earlystopping
240  sumErrorWindow = 0.0;
241  numErrorWindow = 0;
242  currentMin = 1.0;
243  _currentMinT = 0;
244 
245  OUTPUTINFO_OPTIMIZATION opt_type;
246  if (_earlyStopping)
247  {
248  opt_type = pOutInfo->getOutputInfoObject(_earlyStoppingOutputColumn)->getOptimType();
249  if (opt_type == OUTPUTINFO_OPTIMIZATION::UNKNOWN)
250  {
251  cerr << "ERROR!"
252  << _earlyStoppingOutputColumn
253  << " cannot be selected for earlyStopping policy " << endl;
254  exit(1);
255 
256  }
257  }
258 
259  // perform the resuming if necessary. If not it will just return
260  resumeProcess(ss, pTrainingData, pTestData, pOutInfo);
261 
262  if (_verbose == 1)
263  cout << "Learning in progress..." << endl;
264 
265  //I put here the starting time, but it may take very long time to load the saved model
266  time_t startTime, currentTime;
267  time(&startTime);
268 
270  // Starting the AdaBoost main loop
272 // _currentMinT = startingIteration; // early stopping
273 
274  if (_earlyStoppingDone)
275  startingIteration = _numIterations;
276  for (int t = startingIteration; t < _numIterations; ++t)
277  {
278  if (_verbose > 1)
279  cout << "------- WORKING ON ITERATION " << (t+1) << " -------" << endl;
280 
281  BaseLearner* pWeakHypothesis = pWeakHypothesisSource->create();
282  pWeakHypothesis->initLearningOptions(args);
283  //pTrainingData->clearIndexSet();
284 
285  pWeakHypothesis->setTrainingData(pTrainingData);
286 
287  AlphaReal energy = pWeakHypothesis->run();
288 
289  //float gamma = pWeakHypothesis->getEdge();
290  //cout << gamma << endl;
291 
292  if ( (_withConstantLearner) || ( energy != energy ) ) // check constant learner if user wants it (if energi is nan, then we chose constant learner
293  {
294  BaseLearner* pConstantWeakHypothesis = pConstantWeakHypothesisSource->create() ;
295  pConstantWeakHypothesis->initLearningOptions(args);
296  pConstantWeakHypothesis->setTrainingData(pTrainingData);
297  AlphaReal constantEnergy = pConstantWeakHypothesis->run();
298 
299  if ( (constantEnergy <= energy) || ( energy != energy ) || ( nor_utils::is_zero(constantEnergy - energy))) {
300  delete pWeakHypothesis;
301  pWeakHypothesis = pConstantWeakHypothesis;
302  }
303  }
304 
305  if (_verbose > 1)
306  cout << "Weak learner: " << pWeakHypothesis->getName()<< endl;
307 
308 
309  // Output the step-by-step information
310  if (pOutInfo)
311  printOutputInfo(pOutInfo, t, pTrainingData, pTestData, pWeakHypothesis);
312 
313  // Updates the weights and returns the edge
314  AlphaReal gamma = updateWeights(pTrainingData, pWeakHypothesis);
315 
316  if (_verbose > 1)
317  {
318  cout << setprecision(5)
319  << "--> Alpha = " << pWeakHypothesis->getAlpha() << endl
320  << "--> Edge = " << gamma << endl
321  << "--> Energy = " << energy << endl
322  // << "--> ConstantEnergy = " << constantEnergy << endl
323  // << "--> difference = " << (energy - constantEnergy) << endl
324  ;
325  }
326 
327  // If gamma <= theta the algorithm must stop.
328  // If theta == 0 and gamma is 0, it means that the weak learner is no better than chance
329  // and no further training is possible.
330  if (gamma <= _theta)
331  {
332  if (_verbose > 0)
333  {
334  cout << "Can't train any further: edge = " << gamma
335  << " (with and edge offset (theta)=" << _theta << ")" << endl;
336  }
337 
338  // delete pWeakHypothesis;
339  // break;
340  }
341 
342  // append the current weak learner to strong hypothesis file,
343  // that is, serialize it.
344  ss.appendHypothesis(t, pWeakHypothesis);
345 
346  // Add it to the internal list of weak hypotheses
347  _foundHypotheses.push_back(pWeakHypothesis);
348  if (_earlyStopping && !_earlyStoppingDone)
349  {
350  sumErrorWindow += pOutInfo->getOutputHistory(pTestData, _earlyStoppingOutputColumn, t);
351  numErrorWindow += 1;
352  while (numErrorWindow > _earlyStoppingSmoothingWindowRate * t + 1)
353  {
354  sumErrorWindow -= pOutInfo->getOutputHistory(pTestData,_earlyStoppingOutputColumn, t - numErrorWindow + 1);
355  numErrorWindow -= 1;
356  }
357  if (t > _earlyStoppingMinIterations)
358  {
359  if (((opt_type == OUTPUTINFO_OPTIMIZATION::MIN) && ((sumErrorWindow / numErrorWindow) < currentMin)) ||
360  ((opt_type == OUTPUTINFO_OPTIMIZATION::MAX) && ((sumErrorWindow / numErrorWindow) > currentMin)))
361  {
362  currentMin = sumErrorWindow / numErrorWindow;
363  _currentMinT = t;
364  }
365  cout << _earlyStoppingMinIterations << " " << t << ": " << sumErrorWindow/numErrorWindow << " " << _currentMinT << endl;
366  if (t > _currentMinT * _earlyStoppingMaxLookaheadRate)
367  {
368  cout << "Early Stopping at " << t << endl;
369  _earlyStoppingDone = true;
370  break;
371  }
372  }
373  }
374 
375  // check if the time limit has been reached
376  if (_maxTime > 0)
377  {
378  time( &currentTime );
379  float diff = difftime(currentTime, startTime); // difftime is in seconds
380  diff /= 60; // = minutes
381 
382  if (diff > _maxTime)
383  {
384  if (_verbose > 0)
385  cout << "Time limit of " << _maxTime
386  << " minutes has been reached!" << endl;
387  break;
388  }
389  } // check for maxtime
390  delete pWeakHypothesis;
391  } // loop on iterations
393 
394  // write the footer of the strong hypothesis file
395  ss.writeFooter();
396 
397  // write the weights of the instances if the name of weights file isn't empty
398  printOutWeights( pTrainingData );
399 
400 
401  // Free the two input data objects
402  if (pTrainingData)
403  delete pTrainingData;
404  if (pTestData)
405  delete pTestData;
406 
407  if (pOutInfo)
408  delete pOutInfo;
409 
410  if (_verbose > 0)
411  cout << "Learning completed." << endl;
412  }

482  AlphaReal AdaBoostMHLearner::updateWeights(OutputInfo* pOutInfo, InputData* pData, vector<BaseLearner*>& pWeakHypothesiss){
483  const int numExamples = pData->getNumExamples();
484  const int numClasses = pData->getNumClasses();
485 
486  // _hy will contain the margins
487  _hy.resize(numExamples);
488  for ( int i = 0; i < numExamples; ++i){
489  _hy[i].resize(numClasses);
490  fill( _hy[i].begin(), _hy[i].end(), 0.0 );
491 
492  vector<Label>& labels = pData->getLabels(i);
493  // initializing to log weights
494  for (vector<Label>::iterator lIt = labels.begin(); lIt != labels.end(); ++lIt )
495  {
496  lIt->weight = log(lIt->weight);
497  }
498  }
499 
500 
501  if (_verbose > 0)
502  cout << ": 0%." << flush;
503 
504  const int numIters = static_cast<int>(_foundHypotheses.size());
505  const int step = numIters < 5 ? 1 : numIters / 5;
506 
507  vector<BaseLearner*>::iterator it;
508  int t;
509  for(t = 0, it = pWeakHypothesiss.begin(); it != pWeakHypothesiss.end(); it++, t++)
510  {
511  if (_verbose > 1)
512  {
513  if ((t + 1) % 1000 == 0)
514  cout << "." << flush;
515  if ((t + 1) % step == 0)
516  {
517  float progress = static_cast<float>(t) / static_cast<float>(numIters) * 100.0;
518  cout << "." << setprecision(2) << progress << "%." << flush;
519  }
520  }
521 
522  BaseLearner* pWeakHypothesis = *it;
523  const AlphaReal alpha = pWeakHypothesis->getAlpha();
524  AlphaReal hx;
525  for (int i = 0; i < numExamples; ++i)
526  {
527  vector<Label>& labels = pData->getLabels(i);
528  // accumulating margins and log weights
529  for (vector<Label>::iterator lIt = labels.begin(); lIt != labels.end(); ++lIt )
530  {
531  hx = pWeakHypothesis->classify(pData, i, lIt->idx ); // h_l(x_i)
532  _hy[i][lIt->idx] += alpha * hx; // alpha * h_l(x_i)
533  lIt->weight -= alpha * hx * lIt->y; // log(exp( -alpha * h_l(x_i) ))
534  }
535  }
536  }
537 
538 
539  // centering the log weights for avoiding numerical problems
540  AlphaReal meanLogWeights = 0; // the mean of the log weights
541  int numLabels = 0; // the number of the labels (should be counted since the label vector is of variable size)
542  for (int i = 0; i < numExamples; ++i)
543  {
544  vector<Label>& labels = pData->getLabels(i);
545  for (vector<Label>::iterator lIt = labels.begin(); lIt != labels.end(); ++lIt )
546  {
547  meanLogWeights += lIt->weight;
548  numLabels++;
549  }
550  }
551  meanLogWeights /= numLabels;
552 
553  // computing the normalization factor
554  AlphaReal Z = 0; // the normalization factor
555  for (int i = 0; i < numExamples; ++i)
556  {
557  vector<Label>& labels = pData->getLabels(i);
558  for (vector<Label>::iterator lIt = labels.begin(); lIt != labels.end(); ++lIt )
559  {
560  lIt->weight -= meanLogWeights;
561  Z += exp(lIt->weight);
562  }
563  }
564 
565  // normalizing and exponentiating the weights
566  for (int i = 0; i < numExamples; ++i)
567  {
568  vector<Label>& labels = pData->getLabels(i);
569  for (vector<Label>::iterator lIt = labels.begin(); lIt != labels.end(); ++lIt )
570  {
571  lIt->weight = exp(lIt->weight)/Z;
572  }
573  }
574 
575  //upload the margins
576  pOutInfo->setTable( pData, _hy );
577  pOutInfo->setStartingIteration(numIters);
578  return 0;
579  }

584  AlphaReal AdaBoostMHLearner::updateWeights(InputData* pData, BaseLearner* pWeakHypothesis)
585  {
586  const int numExamples = pData->getNumExamples();
587  const int numClasses = pData->getNumClasses();
588 
589  const AlphaReal alpha = pWeakHypothesis->getAlpha();
590 
591  AlphaReal Z = 0; // The normalization factor
592 
593  _hy.resize(numExamples);
594  for ( int i = 0; i < numExamples; ++i) {
595  _hy[i].resize(numClasses);
596  fill( _hy[i].begin(), _hy[i].end(), 0.0 );
597  }
598  // recompute weights
599  // computing the normalization factor Z
600 
601  // for each example
602  for (int i = 0; i < numExamples; ++i)
603  {
604  vector<Label>& labels = pData->getLabels(i);
605  vector<Label>::iterator lIt;
606 
607  for (lIt = labels.begin(); lIt != labels.end(); ++lIt )
608  {
609  _hy[i][lIt->idx] = pWeakHypothesis->classify(pData, i, lIt->idx) * // h_l(x_i)
610  lIt->y;
611  Z += lIt->weight * // w
612  exp(
613  -alpha * _hy[i][lIt->idx] // -alpha * h_l(x_i) * y_i
614  );
615  // important!
616  // _hy[i] must be a vector with different sizes, depending on the
617  // example!
618  // so it will become:
619  // _hy[i][l]
620  // where l is NOT the index of the label (lIt->idx), but the index in the
621  // label vector of the example
622  }
623  }
624 
625  AlphaReal gamma = 0;
626 
627  // Now do the actual re-weight
628  // (and compute the edge at the same time)
629  // for each example
630  for (int i = 0; i < numExamples; ++i)
631  {
632  vector<Label>& labels = pData->getLabels(i);
633  vector<Label>::iterator lIt;
634 
635  for (lIt = labels.begin(); lIt != labels.end(); ++lIt )
636  {
637  AlphaReal w = lIt->weight;
638  gamma += w * _hy[i][lIt->idx];
639  //if ( gamma < -0.8 ) {
640  // cout << gamma << endl;
641  //}
642  // The new weight is w * exp( -alpha * h(x_i) * y_i ) / Z
643  lIt->weight = w * exp( -alpha * _hy[i][lIt->idx] ) / Z;
644  }
645  }
646

649  //{
650  // for (int l = 0; l < numClasses; ++l)
651  // {
652  // _hy[i][l] = pWeakHypothesis->classify(pData, i, l) * // h_l(x_i)
653  // pData->getLabel(i, l); // y_i
654 
655  // Z += pData->getWeight(i, l) * // w
656  // exp(
657  // -alpha * _hy[i][l] // -alpha * h_l(x_i) * y_i
658  // );
659  // } // numClasses
660  //} // numExamples
661 
662  // The edge. It measures the
663  // accuracy of the current weak hypothesis relative to random guessing
664 
667  //for (int i = 0; i < numExamples; ++i)
668  //{
669  // for (int l = 0; l < numClasses; ++l)
670  // {
671  // float w = pData->getWeight(i, l);
672 
673  // gamma += w * _hy[i][l];
674

727  ss.appendHypothesis(t, pWeakHypothesis);

731  // Updates the weights

739  if (pTestData)

743  updateWeights(pOutInfo, pTestData, _foundHypotheses);

747  // Output the step-by-step information
748  //printOutputInfo(pOutInfo, _foundHypotheses.size(), pTrainingData, pTestData, pWeakHypothesis);
749  } else { //slow resume process, in this case the AdaBoost will recalculate the error rates of all iterations

751  const int step = numIters < 5 ? 1 : numIters / 5;
752 
753  OUTPUTINFO_OPTIMIZATION opt_type;

772  BaseLearner* pWeakHypothesis = *it;

819  _earlyStoppingDone = true;
820  break;
821  }
822  }
823  }
824 
825 
826  } // loop on iterations
827  }

903  BaseLearner::RegisteredLearners().getLearner(baseLearnerName);
