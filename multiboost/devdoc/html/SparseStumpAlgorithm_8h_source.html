194  template <typename T>
195  void SparseStumpAlgorithm<T>::findSingleThresholdWithInit
196  (const vpIterator& dataBegin,
197  const vpIterator& dataEnd,
198  InputData* pData,
199  vector<FeatureReal>& thresholds,
200  AlphaReal halfTheta,
201  vector<sRates>* pMu, vector<AlphaReal>* pV)
202  {
203  const int numClasses = pData->getNumClasses();
204 
205  vpIterator currentSplitPos1; // the iterator of the currently examined example
206  vpIterator previousSplitPos1; // the iterator of the example before the current example
207  vpIterator bestSplitPos1; // the iterator of the best split
208  vpIterator bestPreviousSplitPos1; // the iterator of the example before the best split
209 
210  vpIterator currentSplitPos2; // the iterator of the currently examined example
211  vpIterator previousSplitPos2; // the iterator of the example before the current example
212  vpIterator bestSplitPos2; // the iterator of the best split
213  vpIterator bestPreviousSplitPos2; // the iterator of the example before the best split
214 
215 
216  // initialize halfEdges to the constant classifier's half edges
217  copy(_constantEdges.begin(), _constantEdges.end(), _edges.begin());
218 
219  //store the thresholds
220  thresholds.resize(2);
221  thresholds[0] = thresholds[1] = 0.0;
222 
223  AlphaReal currEdge = 0;
224  AlphaReal bestEdge = -numeric_limits<AlphaReal>::max();
225  vector<Label>::const_iterator lIt;
226  vector<AlphaReal> tmpEdges(numClasses);
227 
228  // find the best threshold (cutting point)
229  // at the first split we have
230  // first split: x | x x x x x x x x ..
231  // previous -^ ^- current
232  for( currentSplitPos1 = previousSplitPos1 = dataBegin, ++currentSplitPos1;
233  currentSplitPos1 != dataEnd;
234  previousSplitPos1 = currentSplitPos1, ++currentSplitPos1)
235  {
236  vector<Label>& labels1 = pData->getLabels(previousSplitPos1->first);
237 
238  // recompute edges at the next point
240  for (lIt = labels1.begin(); lIt != labels1.end(); ++lIt )
241  _edges[ lIt->idx ] -= lIt->weight * lIt->y;
243 
244  copy(_edges.begin(), _edges.end(), tmpEdges.begin());
245 
246  // points with the same value of data: to skip because we cannot find a cutting point here!
247  // so we only do the cutting if there is a "hole":
248  if ( previousSplitPos1->second != currentSplitPos1->second )
249  {
250  for( currentSplitPos2 = previousSplitPos2 = previousSplitPos1, ++currentSplitPos2;
251  currentSplitPos2 != dataEnd;
252  previousSplitPos2 = currentSplitPos2, ++currentSplitPos2)
253  {
254  vector<Label>& labels2 = pData->getLabels(previousSplitPos2->first);
255 
256  // recompute edges at the next point
258  for (lIt = labels2.begin(); lIt != labels2.end(); ++lIt )
259  tmpEdges[ lIt->idx ] -= lIt->weight * lIt->y;
261 
262  if ( previousSplitPos2->second != currentSplitPos2->second )
263  {
264  currEdge = 0;
265 
267  for (int l = 0; l < numClasses; ++l) {
268  // flip the class-wise edge if it is negative
269  // but store the flipping bit only at the end (below**)
270  if ( tmpEdges[l] > 0 )
271  currEdge += tmpEdges[l];
272  else
273  currEdge -= tmpEdges[l];
274  }
276 
277  // the current edge is the new maximum
278  if (currEdge > bestEdge)
279  {
280  bestEdge = currEdge;
281  bestSplitPos1 = currentSplitPos1;
282  bestPreviousSplitPos1 = previousSplitPos1;
283  bestSplitPos2 = currentSplitPos2;
284  bestPreviousSplitPos2 = previousSplitPos2;
285 
286  for (int l = 0; l < numClasses; ++l)
287  _bestEdges[l] = tmpEdges[l];
288  }
289  } //endif
290  } // endfor
291  } // endif
292  } //endfor
293 
294  // If we found a valid stump in this dimension
295  if (bestEdge > -numeric_limits<AlphaReal>::max())
296  {
297  thresholds[0] = static_cast<FeatureReal>( bestPreviousSplitPos1->second +
298  bestSplitPos1->second ) / 2;
299 
300  thresholds[1] = static_cast<FeatureReal>( bestPreviousSplitPos2->second +
301  bestSplitPos2->second ) / 2;
302 
303  // Fill the mus if present. This could have been done in the threshold loop,
304  // but here is done just once
305  if ( pMu )
306  {
307  for (int l = 0; l < numClasses; ++l)
308  {
309  // **here
310  if (_bestEdges[l] > 0)
311  (*pV)[l] = +1;
312  else
313  (*pV)[l] = -1;
314 
315  (*pMu)[l].classIdx = l;
316 
317  (*pMu)[l].rPls = _weightsPerClass[l] + (*pV)[l] * _bestEdges[l];
318  (*pMu)[l].rMin = _weightsPerClass[l] - (*pV)[l] * _bestEdges[l];
319  (*pMu)[l].rZero = (*pMu)[l].rPls + (*pMu)[l].rMin; // == weightsPerClass[l]
320  }
321  }
322  //cout << 2 * bestHalfEdge << endl << flush;
323  }
324 
325  } // end of findSingleThresholdWithInit
