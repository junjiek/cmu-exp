81  string next_token()

83  string result;
84  char ch;
85  insert_iterator<string> iIt(result, result.begin());
86 
87  if ( p != end_of_stream)

92  ch = *p;
93  p++;
94  }
95

98  while ( p != end_of_stream && !is_delimiter(*p) )
99  {
100  ch = *p;
101  *iIt++ = *p++;
102  }
103  }
104

106  }
