49  void ConstantAlgorithm::findConstantWeightsEdges( InputData* pData,
50  vector<AlphaReal>& halfWeightsPerClass,
51  vector<AlphaReal>& halfEdges)
52  {
53  const int numClasses = pData->getNumClasses();
54  const int numExamples = pData->getNumExamples();
55 
56  memset(&(halfWeightsPerClass[0]), 0, sizeof(AlphaReal) * halfWeightsPerClass.size() );
57  memset(&(halfEdges[0]), 0, sizeof(AlphaReal) * halfEdges.size() );
58  //fill(halfWeightsPerClass.begin(), halfWeightsPerClass.end(), 0);
59  //fill(halfEdges.begin(), halfEdges.end(), 0);
60 
61  for (int i = 0; i < numExamples; ++i)
62  {
63  vector<Label>& labels = pData->getLabels(i);
64  vector<Label>::iterator lIt;
65 
66  int l = 0;
67  for (lIt = labels.begin(); lIt != labels.end(); ++lIt)
68  {
69  halfWeightsPerClass[ lIt->idx ] += lIt->weight;
70  halfEdges[ lIt->idx ] += lIt->weight * lIt->y;
71  }
72  }
73 
74  for (int l = 0; l < numClasses; ++l)
75  {
76  halfWeightsPerClass[l] /= 2.0;
77  halfEdges[l] /= 2.0;
78  }
79  } // end of findConstantWeightsEdges

83  AlphaReal ConstantAlgorithm::findConstant(InputData* pData,
84  vector<sRates>* pMu, vector<AlphaReal>* pV)
85  {
86  const int numClasses = pData->getNumClasses();
87 
88  vector<AlphaReal> halfWeightsPerClass(numClasses);
89  vector<AlphaReal> halfEdges(numClasses);
90 
91  findConstantWeightsEdges(pData, halfWeightsPerClass, halfEdges);
92 
93  AlphaReal halfEdge = 0;
94 
95  for (int l = 0; l < numClasses; ++l)
96  {
97  if (halfEdges[l] > 0)
98  (*pV)[l] = +1;
99  else
100  (*pV)[l] = -1;
101 
102  halfEdge += (*pV)[l] * halfEdges[l];
103 
104  (*pMu)[l].classIdx = l;
105 
106  (*pMu)[l].rPls = halfWeightsPerClass[l] + (*pV)[l] * halfEdges[l];
107  (*pMu)[l].rMin = halfWeightsPerClass[l] - (*pV)[l] * halfEdges[l];
108  (*pMu)[l].rZero = (*pMu)[l].rPls + (*pMu)[l].rMin; // == weightsPerClass[l]
109  }
110 
111  return 2 * halfEdge;
112 
113  } // end of findConstant
