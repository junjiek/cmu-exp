51  AlphaReal SingleSparseStumpLearner::run()
52  {
53  const int numClasses = _pTrainingData->getNumClasses();
54  const int numColumns = _pTrainingData->getNumAttributes();
55 
56  // set the smoothing value to avoid numerical problem
57  // when theta=0.
58  setSmoothingVal( 1.0 / (AlphaReal)_pTrainingData->getNumExamples() * 0.01 );
59 
60  vector<sRates> mu(numClasses); // The class-wise rates. See BaseLearner::sRates for more info.
61  vector<AlphaReal> tmpV(numClasses); // The class-wise votes/abstentions
62 
63  FeatureReal tmpThreshold;
64  AlphaReal tmpAlpha;
65 
66  AlphaReal bestEnergy = numeric_limits<AlphaReal>::max();
67  AlphaReal tmpEnergy;
68 
69  StumpAlgorithmLSHTC<FeatureReal> sAlgo(numClasses);
70  sAlgo.initSearchLoop(_pTrainingData);
71 
72  AlphaReal halfTheta;
73  if ( _abstention == ABST_REAL || _abstention == ABST_CLASSWISE )
74  halfTheta = _theta/2.0;
75  else
76  halfTheta = 0;
77 
78  int numOfDimensions = _maxNumOfDimensions;
79  for (int j = 0; j < numColumns; ++j)
80  {
81  // Tricky way to select numOfDimensions columns randomly out of numColumns
82  int rest = numColumns - j;
83  float r = rand()/static_cast<float>(RAND_MAX);
84 
85  if ( static_cast<float>(numOfDimensions) / rest > r )
86  {
87  --numOfDimensions;
88 
89  //if ( static_cast<SortedData*>(_pTrainingData)->isAttributeEmpty( j ) ) continue;
90 
91  const pair<vpReverseIterator,vpReverseIterator> dataBeginEnd =
92  static_cast<SortedData*>(_pTrainingData)->getFilteredReverseBeginEnd(j);
93 
94  //if ( static_cast<SortedData*>(_pTrainingData)->isFilteredAttributeEmpty() ) continue;
95  //if ( static_cast<SortedData*>(_pTrainingData)->isFilteredAttributeHasOneValue() ) continue;
96 
97  const vpReverseIterator dataBegin = dataBeginEnd.first;
98  const vpReverseIterator dataEnd = dataBeginEnd.second;
99 
100 
101 
102  // also sets mu, tmpV, and bestHalfEdge
103  tmpThreshold = sAlgo.findSingleThresholdWithInit(dataBegin, dataEnd, _pTrainingData,
104  halfTheta, &mu, &tmpV);
105 
106  if (tmpThreshold == tmpThreshold) // tricky way to test Nan
107  {
108  // small inconsistency compared to the standard algo (but a good
109  // trade-off): in findThreshold we maximize the edge (suboptimal but
110  // fast) but here (among dimensions) we minimize the energy.
111  tmpEnergy = getEnergy(mu, tmpAlpha, tmpV);
112 
113  if (tmpEnergy < bestEnergy && tmpAlpha > 0)
114  {
115  // Store it in the current weak hypothesis.
116  // note: I don't really like having so many temp variables
117  // but the alternative would be a structure, which would need
118  // to be inheritable to make things more consistent. But this would
119  // make it less flexible. Therefore, I am still undecided. This
120  // might change!
121 
122  _alpha = tmpAlpha;
123  _v = tmpV;
124  _selectedColumn = j;
125  _threshold = tmpThreshold;
126 
127  bestEnergy = tmpEnergy;
128  }
129  } // tmpThreshold == tmpThreshold
130  }
131  }
132 
133  if ( _selectedColumn != -1 )
134  {
135  stringstream thresholdString;
136  thresholdString << _threshold;
137  _id = _pTrainingData->getAttributeNameMap().getNameFromIdx(_selectedColumn) + thresholdString.str();
138  } else {
139  bestEnergy = numeric_limits<AlphaReal>::signaling_NaN();
140  }
141 
142  return bestEnergy;
143 
144  }
