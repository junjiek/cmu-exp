45  void TxtParser::readData( vector<Example>& examples, NameMap& classMap,
46  vector<NameMap>& enumMaps, NameMap& attributeNameMap,
47  vector<RawData::eAttributeType>& attributeTypes )
48  {
49  // open file
50  ifstream inFile(_fileName.c_str());
51  if ( !inFile.is_open() )
52  {
53  cerr << "\nERROR: Cannot open file <" << _fileName << ">!!" << endl;
54  exit(1);
55  }
56 
57  // set white spaces to consider tab as NOT whitespace
58  // the white_tab will be erased automatically by fstream
59  inFile.imbue( locale(locale(), new nor_utils::white_spaces(_sepChars) ) );
60 
61  _numAttributes = (int)nor_utils::count_columns(inFile);
62 
63  // if it has a filename for each example, don't count it
64  if (_hasExampleName)
65  --_numAttributes;
66 
67  // the class is not a data column
68  --_numAttributes;
69 
70  string line;
71  getline( inFile, line );
72  if ( !checkInput( line, _numAttributes ) )
73  {
74  cerr << "\nERROR: Input file not correct, check file <" << _fileName << "> for errors," << endl
75  << "or your separation option -d (if you are using it)" << endl;
76  exit(1);
77  }
78 
79  inFile.clear(); // reset position
80  inFile.seekg(0);
81 
82  string tmpExampleName;
83  string tmpClassName;
84  vector<string> tmpClassStored;
85  if (_verboseLevel > 0) cout << "Counting rows.." << flush;
86  size_t numRows = nor_utils::count_rows(inFile);
87 
88  if (_verboseLevel > 0) cout << "Allocating.." << flush;
89  try {
90  examples.resize(numRows);
91  tmpClassStored.resize(numRows);
92  }
93  catch(...) {
94  cerr << "ERROR: Cannot allocate memory for storage!" << endl;
95  exit(1);
96  }
97 
98  if (_verboseLevel > 0) cout << "Done!" << endl;
99 
100  if (_verboseLevel > 0) cout << "Reading file.." << endl;
101 
103  size_t i;
104  for (i = 0; i < numRows; ++i)
105  {
106  if (_hasExampleName)
107  inFile >> tmpExampleName; // store file name
108 
109  if (!_hasClassEnd)
110  inFile >> tmpClassName; // store class
111 
112  Example& currExample = examples[i];
113  currExample.setName(tmpExampleName);
114 
115  vector<FeatureReal>& values = currExample.getValues();
116  values.resize(_numAttributes);
117 
118  for (int j = 0; j < _numAttributes; ++j)
119  inFile >> values[j]; // store values
120 
121  // to avoid problems in the case of an empty line at the end
122  // of the file
123  if ( inFile.eof() )
124  break;
125 
126  if (_hasClassEnd)
127  inFile >> tmpClassName; // store class
128 
129  int classIdx = classMap.addName(tmpClassName);
130  tmpClassStored[i] = tmpClassName;
131  //if ( classIdx > 1 )
132  //{
133  // cerr << "ERROR: Only binary labels are accepted (max 2!) for .txt parser!!" << endl;
134  // exit(1);
135  //}
136 // currExample.addBinaryLabel(classIdx);
137  }
138 
139  if ( i != examples.size() )
140  {
141  cerr << "WARNING: Different number of read examples ("
142  << i << ") and lines ("
143  << examples.size() << ")!" << endl;
144 
145  examples.resize(i);
146  }
147 
149  for (int i = 0; i < _numAttributes; i++)
150  {
151  std::ostringstream oss;
152  oss << "a" << i;
153  attributeNameMap.addName(oss.str());
154  attributeTypes.push_back(RawData::ATTRIBUTE_NUMERIC);
155  }
156 
157  const int numClasses = classMap.getNumNames();
158  for (int i = 0; i < numRows; i++)
159  {
160  vector<Label> labels;
161  labels.resize(numClasses);
162  for ( int j = 0; j < numClasses; ++j )
163  {
164  labels[j].idx = j;
165  labels[j].y = -1;
166  }
167  labels[ classMap.getIdxFromName(tmpClassStored[i]) ].y = +1;
168  examples[i].addLabels(labels);
169  }
170  if (_verboseLevel > 0) cout << "Done!" << endl;
171  }

175  bool TxtParser::checkInput(const string& line, int numColumns) const
176  {
177  istringstream ss(line);
178  ss.imbue( locale(locale(), new nor_utils::white_spaces(_sepChars) ) );
179 
180  string tmp;
181  bool inputValid = true;
182 
183  if (_hasExampleName)
184  ss >> tmp; // example name
185 
186  if (!_hasClassEnd)
187  ss >> tmp; // class at the beginning
188 
189  for (int j = 0; j < numColumns; ++j)
190  {
191  if ( ss.eof() )
192  {
193  inputValid = false;
194  break;
195  }
196 
197  ss >> tmp;
198 
199  if ( !nor_utils::is_number(tmp) )
200  {
201  inputValid = false;
202  break;
203  }
204  }
205 
206  if (_hasClassEnd)
207  ss >> tmp; // class
208  if ( tmp.empty() )
209  inputValid = false;
210 
211  return inputValid;
212  }
